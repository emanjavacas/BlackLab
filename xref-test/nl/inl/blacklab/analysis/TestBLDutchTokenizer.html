16  package nl.inl.blacklab.analysis;
17  
18  import java.io.IOException;
19  import java.io.Reader;
20  import java.io.StringReader;
21  
22  import org.apache.lucene.analysis.Tokenizer;
23  import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
24  import org.junit.Assert;
25  import org.junit.Test;
26  
27  public class TestBLDutchTokenizer {
28  
29  	@Test
30  	public void testBasics() throws IOException {
31  		Reader r = new StringReader("\"hond, a.u.b.: bél(len); \t [pre]cursor \t\nzo'n 'Hij zij' ex-man -");
32  		try (Tokenizer ts = new BLDutchTokenizer()) {
33  			ts.setReader(r);
34  			ts.reset();
35  			CharTermAttribute ta = ts.addAttribute(CharTermAttribute.class);
36  			Assert.assertTrue(ts.incrementToken());
37  			Assert.assertEquals("hond", new String(ta.buffer(), 0, ta.length()));
38  			Assert.assertTrue(ts.incrementToken());
39  			Assert.assertEquals("a.u.b.", new String(ta.buffer(), 0, ta.length()));
40  			Assert.assertTrue(ts.incrementToken());
41  			Assert.assertEquals("bél(len)", new String(ta.buffer(), 0, ta.length()));
42  			Assert.assertTrue(ts.incrementToken());
43  			Assert.assertEquals("[pre]cursor", new String(ta.buffer(), 0, ta.length()));
44  			Assert.assertTrue(ts.incrementToken());
45  			Assert.assertEquals("zo'n", new String(ta.buffer(), 0, ta.length()));
46  			Assert.assertTrue(ts.incrementToken());
47  			Assert.assertEquals("'Hij", new String(ta.buffer(), 0, ta.length()));
48  			Assert.assertTrue(ts.incrementToken());
49  			Assert.assertEquals("zij'", new String(ta.buffer(), 0, ta.length()));
50  			Assert.assertTrue(ts.incrementToken());
51  			Assert.assertEquals("ex-man", new String(ta.buffer(), 0, ta.length()));
52  			Assert.assertTrue(ts.incrementToken());
53  			Assert.assertEquals("-", new String(ta.buffer(), 0, ta.length()));
54  			Assert.assertFalse(ts.incrementToken());
55  		}
56  	}
57  
58  }
