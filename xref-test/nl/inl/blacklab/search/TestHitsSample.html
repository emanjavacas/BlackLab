16  package nl.inl.blacklab.search;
17  
18  import org.junit.Assert;
19  import org.junit.Test;
20  
21  import nl.inl.blacklab.MockHits;
22  import nl.inl.blacklab.testutil.MockSearcher;
23  
24  public class TestHitsSample {
25  
26  	private final static int[] aDoc   = new int[] { 1, 1, 2, 3, 3, 3 };
27  	private final static int[] aStart = new int[] { 1, 4, 2, 1, 3, 5 };
28  	private final static int[] aEnd   = new int[] { 2, 5, 3, 2, 4, 7 };
29  
30  	private static void assertSampleRatio(int[] expected, float ratio, long seed) {
31  		Hits hits = HitsSample.fromHits(new MockHits(new MockSearcher(), aDoc, aStart, aEnd), ratio, seed);
32  		int i = 0;
33  		Assert.assertEquals(expected.length, hits.size());
34  		for (Hit hit: hits) {
35  			Assert.assertEquals(aDoc[expected[i]],   hit.doc);
36  			Assert.assertEquals(aStart[expected[i]], hit.start);
37  			Assert.assertEquals(aEnd[expected[i]],   hit.end);
38  			i++;
39  		}
40  	}
41  
42  	private static void assertSampleNumber(int[] expected, int number, long seed) {
43  		Hits hits = HitsSample.fromHits(new MockHits(new MockSearcher(), aDoc, aStart, aEnd), number, seed);
44  		int i = 0;
45  		Assert.assertEquals(expected.length, hits.size());
46  		for (Hit hit: hits) {
47  			Assert.assertEquals(aDoc[expected[i]],   hit.doc);
48  			Assert.assertEquals(aStart[expected[i]], hit.start);
49  			Assert.assertEquals(aEnd[expected[i]],   hit.end);
50  			i++;
51  		}
52  	}
53  
54  	@Test
55  	public void testSample() {
56  		assertSampleRatio(new int[0], 0, 0);
57  		assertSampleRatio(new int[] {3}, 0.1667f, 1);
58  		assertSampleRatio(new int[] {0, 1, 5}, 0.5f, 1337);
59  		assertSampleRatio(new int[] {0, 1, 2, 3}, 0.6667f, 42);
60  		assertSampleRatio(new int[] {0, 1, 2, 3, 4, 5}, 1f, Long.MAX_VALUE);
61  
62  		assertSampleNumber(new int[0], 0, 0);
63  		assertSampleNumber(new int[] {3}, 1, 1);
64  		assertSampleNumber(new int[] {0, 1, 5}, 3, 1337);
65  		assertSampleNumber(new int[] {0, 1, 2, 3}, 4, 42);
66  		assertSampleNumber(new int[] {0, 1, 2, 3, 4, 5}, 6, Long.MAX_VALUE);
67  	}
68  
69  }
