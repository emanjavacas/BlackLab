20  public class TestHitPropertySerialize {
21  
22  	MockSearcher mockSearcher = new MockSearcher();
23  
24  	Hits hits = new MockHits(mockSearcher);
25  
26  	@Before
27  	public void setUp() {
28  		mockSearcher.setForwardIndex("contents%lemma", new MockForwardIndex(new MockTerms("aap", "noot", "mies")));
29  	}
30  
31  	@Test
32  	public void testHitPropertySerialize() {
33  		HitProperty prop;
34  
35  		prop = new HitPropertyDocumentDecade(hits, "decade");
36  		Assert.assertEquals("decade:decade", prop.serialize());
37  
38  		prop = new HitPropertyDocumentId(hits);
39  		prop.setReverse(true);
40  		String exp = "-docid";
41  		Assert.assertEquals(exp, prop.serialize());
42  		Assert.assertEquals(exp, HitProperty.deserialize(hits, exp).serialize());
43  
44  		prop = new HitPropertyHitText(hits, "contents", "lemma", true);
45  		Assert.assertEquals("hit:lemma:s", prop.serialize());
46  
47  		List<ContextPart> contextParts = Arrays.asList(
48  			new ContextPart(ContextStart.LEFT_OF_HIT, 1, 1),         // second word to left of hit
49  			new ContextPart(ContextStart.HIT_TEXT_FROM_START, 0, 1), // first two hit words
50  			new ContextPart(ContextStart.HIT_TEXT_FROM_END, 0, 0)    // last hit word
51  		);
52  		prop = new HitPropertyContextWords(hits, "contents", "lemma", true, contextParts);
53  		Assert.assertEquals("context:lemma:s:L2-2;H1-2;E1-1", prop.serialize());
54  	}
55  
56  	@Test
57  	public void testDocPropertySerialize() {
58  		DocProperty prop;
59  
60  		prop = new DocPropertyDecade("decade");
61  		prop.setReverse(true);
62  		String exp = "-decade:decade";
63  		Assert.assertEquals(exp, prop.serialize());
64  		Assert.assertEquals(exp, DocProperty.deserialize(exp).serialize());
65  	}
66  
67  	@Test
68  	public void testHitPropValueSerialize() {
69  		HitPropValue val, val1;
70  
71  		val1 = new HitPropValueContextWord(hits, "lemma", 2, true);
72  		String exp = "cwo:lemma:s:mies";
73  		Assert.assertEquals(exp, val1.serialize());
74  		Assert.assertEquals(exp, HitPropValue.deserialize(hits, exp).serialize());
75  
76  		val1 = new HitPropValueDecade(1980);
77  		exp = "dec:1980";
78  		Assert.assertEquals(exp, val1.serialize());
79  		Assert.assertEquals(exp, HitPropValue.deserialize(hits, exp).serialize());
80  
81  		val = new HitPropValueMultiple(new HitPropValue[] {val1, new HitPropValueString("blabla")});
82  		exp = "dec:1980,str:blabla";
83  		Assert.assertEquals(exp, val.serialize());
84  		Assert.assertEquals(exp, HitPropValue.deserialize(hits, exp).serialize());
85  
86  		val = new HitPropValueMultiple(new HitPropValue[] {val1, new HitPropValueString("$bl:ab,la")});
87  		exp = "dec:1980,str:$DLbl$CLab$CMla";
88  		Assert.assertEquals(exp, val.serialize());
89  		Assert.assertEquals(exp, HitPropValue.deserialize(hits, exp).serialize());
90  	}
