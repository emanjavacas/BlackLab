15  	private final static int NO_TERM = Terms.NO_TERM;

17  	private static TestIndex testIndex;
18  
19  	private static Terms terms;
20  
21  	@BeforeClass
22  	public static void setUp() throws Exception {
23  		testIndex = new TestIndex();
24  		terms = testIndex.getSearcher().getForwardIndex("contents%word").getTerms();
25  	}
26  
27  	private static int term(String word) {
28  		return terms.indexOf(word);
29  	}
30  
31  	@AfterClass
32  	public static void tearDown() {
33  		testIndex.close();
34  	}

36  	@Test
37  	public void testHitPropHitText() throws ParseException {
38  		Hits hits = testIndex.find(" 'the' ");
39  		HitProperty p = new HitPropertyHitText(hits, true);
40  		HitGroups g = hits.groupedBy(p);
41  		HitGroup group = g.getGroup(new HitPropValueContextWords(hits, "word", new int[] {term("the")}, true));
42  		Assert.assertEquals(3, group.size());
43  		group = g.getGroup(new HitPropValueContextWords(hits, "word", new int[] {term("The")}, true));
44  		Assert.assertEquals(1, group.size());
45  	}
46  
47  	@Test
48  	public void testHitPropContextWords() throws ParseException {
49  		Hits hits = testIndex.find(" 'the' ");
50  		HitProperty p = new HitPropertyContextWords(hits, "contents", "word", true, "L1-1;H1-2");
51  		HitGroups g = hits.groupedBy(p);
52  		HitGroup group;
53  		group = g.getGroup(new HitPropValueContextWords(hits, "word", new int[] {NO_TERM, term("The"), NO_TERM}, true));
54  		Assert.assertEquals(1, group.size());
55  		group = g.getGroup(new HitPropValueContextWords(hits, "word", new int[] {term("over"), term("the"), NO_TERM}, true));
56  		Assert.assertEquals(1, group.size());
57  		group = g.getGroup(new HitPropValueContextWords(hits, "word", new int[] {term("May"), term("the"), NO_TERM}, true));
58  		Assert.assertEquals(1, group.size());
59  		group = g.getGroup(new HitPropValueContextWords(hits, "word", new int[] {term("is"), term("the"), NO_TERM}, true));
60  		Assert.assertEquals(1, group.size());
61  	}
