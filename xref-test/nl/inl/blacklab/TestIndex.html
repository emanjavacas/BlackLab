55  	/**
56  	 * The BlackLab searcher object.
57  	 */
58  	Searcher searcher;
59  
60  	private File indexDir;
61  
62  	public TestIndex() throws Exception {
63  		// Get a temporary directory for our test index
64  		indexDir = new File(System.getProperty("java.io.tmpdir"),
65  				"BlackLabExample");
66  		if (indexDir.exists()) {
67  			// Delete the old example dir
68  			// (NOTE: we also try to do this on exit but it may fail due to
69  			// memory mapping (on Windows))
70  			deleteTree(indexDir);
71  		}
72  
73  		// Instantiate the BlackLab indexer, supplying our DocIndexer class
74  		Indexer indexer = new Indexer(indexDir, true, DocIndexerExample.class);
75  		indexer.setListener(new IndexListenerDevNull()); // no output
76  		try {
77  			// Index each of our test "documents".
78  			for (int i = 0; i < testData.length; i++) {
79  				indexer.index("test" + (i + 1), new StringReader(testData[i]));
80  			}
81  		} finally {
82  			// Finalize and close the index.
83  			indexer.close();
84  		}
85  
86  		// Create the BlackLab searcher object
87  		searcher = Searcher.open(indexDir);
88  		searcher.hitsSettings().setContextSize(1);
89  	}
90  
91  	public Searcher getSearcher() {
92  		return searcher;
93  	}
94  
95  	public void close() {
96  		if (searcher != null)
97  			searcher.close();
98  		deleteTree(indexDir);
99  	}

101 	private static void deleteTree(File dir) {
102 		for (File f: dir.listFiles()) {
103 			if (f.isFile())
104 				f.delete();
105 			else if (f.isDirectory())
106 				deleteTree(f);
107 		}
108 		dir.delete();
109 	}

111 	/**
112 	 * Find concordances from a Corpus Query Language query.
113 	 *
114 	 * @param query
115 	 *            the query to parse
116 	 * @return the resulting BlackLab text pattern
117 	 * @throws ParseException
118 	 */
119 	public List<String> findConc(String query) throws ParseException {
120 		Hits hits = find(query);
121 		return getConcordances(hits);
122 	}
123 
124 	/**
125 	 * Find hits from a Corpus Query Language query.
126 	 *
127 	 * @param query
128 	 *            the query to parse
129 	 * @return the resulting BlackLab text pattern
130 	 * @throws ParseException
131 	 */
132 	public Hits find(String query) throws ParseException {
133 		// Parse query using the CorpusQL parser
134 		TextPattern tp = CorpusQueryLanguageParser.parse(query);
135 
136 		// Execute the search
137 		Hits hits = searcher.find(tp);
138 		return hits;
139 	}
140 
141 	/**
142 	 * Return a list of concordance strings.
143 	 *
144 	 * @param hits
145 	 *            the hits to display
146 	 * @return
147 	 */
148 	static List<String> getConcordances(Hits hits) {
149 		List<String> results = new ArrayList<>();
150 		for (Hit hit : hits) {
151 			Kwic kwic = hits.getKwic(hit);
152 			String left = StringUtil.join(kwic.getLeft("word"), " ");
153 			String match = StringUtil.join(kwic.getMatch("word"), " ");
154 			String right = StringUtil.join(kwic.getRight("word"), " ");
155 			String conc = left + " [" + match + "] " + right;
156 			results.add(conc.trim());
157 		}
158 		return results;
159 	}
