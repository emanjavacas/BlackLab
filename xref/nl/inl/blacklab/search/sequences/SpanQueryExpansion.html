16  package nl.inl.blacklab.search.sequences;
17  
18  import java.io.IOException;
19  import java.util.Map;
20  
21  import org.apache.lucene.index.LeafReaderContext;
22  import org.apache.lucene.index.Term;
23  import org.apache.lucene.index.TermContext;
24  import org.apache.lucene.search.spans.SpanQuery;
25  import org.apache.lucene.search.spans.Spans;
26  import org.apache.lucene.util.Bits;
27  
28  import nl.inl.blacklab.search.lucene.BLSpans;
29  import nl.inl.blacklab.search.lucene.SpanQueryBase;
30  import nl.inl.blacklab.search.lucene.SpansUnique;

32  /**
33   * Expands the source spans to the left and right by the given ranges.
34   *
35   * This is used to support sequences including subsequences of completely unknown tokens (like
36   * "apple" []{2, 4} "pear" to find apple and pear with 2 to 4 tokens in between).
37   *
38   * Note that this class will generate all possible expansions, so if you call it with left-expansion
39   * of between 2 to 4 tokens, it will generate 3 new hits for every hit from the source spans: one
40   * hit with 2 more tokens to the left, one hit with 3 more tokens to the left, and one hit with 4
41   * more tokens to the left.
42   *
43   * Spans generated from this query will be sorted by start point and then by end point, and any
44   * duplicates generated will be discarded.
45   */
46  public class SpanQueryExpansion extends SpanQueryBase {
47  
48  	/** Whether to expand to left (true) or right (false) */
49  	private boolean expandToLeft;
50  
51  	/** Minimum number of tokens to expand */
52  	private int min;
53  
54  	/** Maximum number of tokens to expand (-1 = infinite) */
55  	private int max;
56  
57  	/** if true, we assume the last token is always a special closing token and ignore it */
58  	boolean ignoreLastToken = false;
59  
60  	public SpanQueryExpansion(SpanQuery clause, boolean expandToLeft, int min, int max) {
61  		super(clause);
62  		this.expandToLeft = expandToLeft;
63  		this.min = min;
64  		this.max = max;
65  		if (max != -1 && min > max)
66  			throw new IllegalArgumentException("min > max");
67  	}
68  
69  	@Override
70  	public boolean equals(Object o) {
71  		if (this == o)
72  			return true;
73  		if (!super.equals(o))
74  			return false;
75  
76  		final SpanQueryExpansion that = (SpanQueryExpansion) o;
77  		return expandToLeft == that.expandToLeft && min == that.min && max == that.max;
78  	}
79  
80  	@Override
81  	public Spans getSpans(LeafReaderContext context, Bits acceptDocs, Map<Term,TermContext> termContexts) throws IOException {
82  		Spans spansSource = clauses[0].getSpans(context, acceptDocs, termContexts);
83  		if (spansSource == null)
84  			return null;
85  		BLSpans spans = new SpansExpansionRaw(ignoreLastToken, context.reader(), clauses[0].getField(), spansSource, expandToLeft, min, max);
86  
87  		// Note: the spans coming from SpansExpansion are not sorted properly.
88  		// Before returning the final spans, we wrap it in a per-document (start-point) sorter.
89  
90  		// Sort the resulting spans by start point.
91  		// Note that duplicates may have formed by combining spans from left and right. Eliminate
92  		// these duplicates now (hence the 'true').
93  		boolean sorted = spans.hitsStartPointSorted();
94  		boolean unique = spans.hitsAreUnique();
95  		if (!sorted) {
96  			return new PerDocumentSortedSpans(spans, false, !unique);
97  		} else if (!unique) {
98  			return new SpansUnique(spans);
99  		}
100 		return spans;
101 	}
102 
103 	@Override
104 	public int hashCode() {
105 		int h = clauses.hashCode();
106 		h ^= (h << 10) | (h >>> 23);
107 		h ^= min << 10;
108 		h ^= max << 5;
109 		h ^= expandToLeft ? 1 : 0;
110 		h ^= Float.floatToRawIntBits(getBoost());
111 		return h;
112 	}
113 
114 	@Override
115 	public String toString(String field) {
116 		return "SpanQueryExpansion(" + clauses[0] + ", " + expandToLeft + ", " + min + ", " + max
117 				+ ")";
118 	}
119 
120 	/** Set whether to ignore the last token.
121 	 *
122 	 * @param ignoreLastToken if true, we assume the last token is always a special closing token and ignore it
123 	 */
124 	public void setIgnoreLastToken(boolean ignoreLastToken) {
125 		this.ignoreLastToken = ignoreLastToken;
126 	}
127 
128 }
