26  /**
27   * A "keyword in context" for a hit
28   * (left context, hit text, right context).
29   *
30   * The Hits class matches this to the Hit.
31   *
32   * This object may be converted to a Concordance object (with XML strings)
33   * by calling Kwic.toConcordance().
34   */
35  public class Kwic {
36  
37  	DocContentsFromForwardIndex fragment;
38  
39  	int hitStart;
40  
41  	int hitEnd;
42  
43  	/**
44  	 * Construct a Kwic object
45  	 *
46  	 * @param properties
47  	 *            What properties are stored in what order for this Kwic (e.g. word, lemma, pos)
48  	 * @param left
49  	 * @param match
50  	 * @param right
51  	 * @deprecated use contructor that takes a single list of tokens and two offsets
52  	 */
53  	@Deprecated
54  	public Kwic(List<String> properties, List<String> left, List<String> match, List<String> right) {
55  		List<String> tokens = new ArrayList<>();
56  		tokens.addAll(left);
57  		tokens.addAll(match);
58  		tokens.addAll(right);
59  		fragment = new DocContentsFromForwardIndex(properties, tokens);
60  		hitStart = left.size() / properties.size();
61  		hitEnd = hitStart + match.size() / properties.size();
62  	}
63  
64  	/**
65  	 * Construct a Kwic object
66  	 *
67  	 * @param properties
68  	 *            What properties are stored in what order for this Kwic (e.g. word, lemma, pos)
69  	 * @param tokens the contents
70  	 * @param matchStart where the match starts, in word positions
71  	 * @param matchEnd where the match ends, in word positions
72  	 */
73  	public Kwic(List<String> properties, List<String> tokens, int matchStart, int matchEnd) {
74  		fragment = new DocContentsFromForwardIndex(properties, tokens);
75  		this.hitStart = matchStart;
76  		this.hitEnd = matchEnd;
77  	}
78  
79  	/**
80  	 * Construct a Kwic object
81  	 *
82  	 * @param fragment the content fragment to make the Kwic from
83  	 * @param matchStart where the match starts, in word positions
84  	 * @param matchEnd where the match ends, in word positions
85  	 */
86  	public Kwic(DocContentsFromForwardIndex fragment, int matchStart, int matchEnd) {
87  		this.fragment = fragment;
88  		this.hitStart = matchStart;
89  		this.hitEnd = matchEnd;
90  	}

92  	public List<String> getLeft() {
93  		return Collections.unmodifiableList(fragment.tokens.subList(0, hitStart * fragment.properties.size()));
94  	}
95  
96  	public List<String> getMatch() {
97  		return Collections.unmodifiableList(fragment.tokens.subList(hitStart * fragment.properties.size(), hitEnd * fragment.properties.size()));
98  	}
99  
100 	public List<String> getRight() {
101 		return Collections.unmodifiableList(fragment.tokens.subList(hitEnd * fragment.properties.size(), fragment.tokens.size()));
102 	}

123 	/**
124 	 * Get the context of a specific property from the complete
125 	 * context list.
126 	 *
127 	 * @param allContext the complete context list of all properties
128 	 * @param property the property to get the context for
129 	 * @param start first word position to get the property context for
130 	 * @param end word position after the last to get the property context for
131 	 * @return the context for this property
132 	 */
133 	private List<String> getSinglePropertyContext(String property, int start, int end) {
134 		final int nProp = fragment.properties.size();
135 		final int size = end - start;
136 		final int propIndex = fragment.properties.indexOf(property);
137 		final int startIndex = start * nProp + propIndex;
138 		if (propIndex == -1)
139 			return null;
140 		return new AbstractList<String>() {
141 			@Override
142 			public String get(int index) {
143 				if (index >= size)
144 					throw new IndexOutOfBoundsException();
145 				return fragment.tokens.get(startIndex + nProp * index);
146 			}
147 
148 			@Override
149 			public int size() {
150 				return size;
151 			}
152 		};
153 	}
154 
155 	/**
156 	 * Get the left context of a specific property
157 	 * @param property the property to get the context for
158 	 * @return the context
159 	 */
160 	public List<String> getLeft(String property) {
161 		return getSinglePropertyContext(property, 0, hitStart);
162 	}
163 
164 	/**
165 	 * Get the match context of a specific property
166 	 * @param property the property to get the context for
167 	 * @return the context
168 	 */
169 	public List<String> getMatch(String property) {
170 		return getSinglePropertyContext(property, hitStart, hitEnd);
171 	}

173 
174 	/**
175 	 * Get the right context of a specific property
176 	 * @param property the property to get the context for
177 	 * @return the context
178 	 */
179 	public List<String> getRight(String property) {
180 		return getSinglePropertyContext(property, hitEnd, fragment.tokens.size() / fragment.properties.size());
181 	}

183 	/**
184 	 * Convert this Kwic object to a Concordance object.
185 	 *
186 	 * This produces XML consisting of &lt;w&gt; tags. The words
187 	 * are the text content of the tags. The punctuation is between the tags.
188 	 * The other properties are attributes of the tags.
189 	 *
190 	 * @return the Concordance object
191 	 */
192 	public Concordance toConcordance() {
193 		return toConcordance(true);
194 	}
195 
196 	/**
197 	 * Convert this Kwic object to a Concordance object.
198 	 *
199 	 * This may either consist of only words and punctuation, or include the XML
200 	 * tags containing the other properties as well, depending on the parameter.
201 	 *
202 	 * @param produceXml if true, produces XML. If false, produces human-readable text.
203 	 * @return the Concordance object
204 	 */
205 	public Concordance toConcordance(boolean produceXml) {
206 		String[] conc = new String[3];
207 		List<String> match = getMatch();
208 		String addPunctAfter = !match.isEmpty() ? match.get(0) : "";
209 		conc[0] = xmlString(getLeft(), addPunctAfter, true, produceXml);
210 		conc[1] = xmlString(match, null, true, produceXml);
211 		conc[2] = xmlString(getRight(), null, false, produceXml);
212 		return new Concordance(conc);
213 	}
214 
215 	/**
216 	 * Convert a context List to an XML string (used for converting to a Concordance object)
217 	 *
218 	 * @param context the context List to convert
219 	 * @param addPunctAfter if not null, this is appended at the end of the string.
220 	 * @param leavePunctBefore if true, no punctuation is added before the first word.
221 	 * @param produceXml if true, produces XML with word tags. If false, produces human-readable text.
222 	 * @return the XML string
223 	 */
224 	private String xmlString(List<String> context, String addPunctAfter, boolean leavePunctBefore, boolean produceXml) {
225 		int valuesPerWord = fragment.properties.size();
226 		int numberOfWords = context.size() / valuesPerWord;
227 		StringBuilder b = new StringBuilder();
228 		for (int i = 0; i < numberOfWords; i++) {
229 			int vIndex = i * valuesPerWord;
230 			int j = 0;
231 			if (i > 0 || !leavePunctBefore) {
232 				if (produceXml)
233 					b.append(StringUtil.escapeXmlChars(context.get(vIndex)));
234 				else
235 					b.append(context.get(vIndex));
236 			}
237 			if (produceXml) {
238 				b.append("<w");
239 				for (int k = 1; k < valuesPerWord - 1; k++) {
240 					String name = fragment.properties.get(k);
241 					String value = context.get(vIndex + 1 + j);
242 					b.append(" ").append(name).append("=\"").append(StringUtil.escapeXmlChars(value)).append("\"");
243 					j++;
244 				}
245 				b.append(">");
246 			} else {
247 				// We're skipping the other properties besides word and punct. Advance j.
248 				if (valuesPerWord > 2)
249 					j += valuesPerWord - 2;
250 			}
251 			if (produceXml)
252 				b.append(StringUtil.escapeXmlChars(context.get(vIndex + 1 + j))).append("</w>");
253 			else
254 				b.append(context.get(vIndex + 1 + j));
255 		}
256 		if (addPunctAfter != null)
257 			b.append(addPunctAfter);
258 		return b.toString();
259 	}
