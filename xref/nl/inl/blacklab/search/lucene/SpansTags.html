30  /**
31   * Gets spans for a certain XML element.
32   */
33  class SpansTags extends BLSpans {
34  	private static final int ARRAYLIST_REALLOC_THRESHOLD = 1000;
35  
36  	static SpanComparatorStartPoint cmpStartPoint = new SpanComparatorStartPoint();
37  
38  	/** The two sets of hits to combine */
39  	private SpansInBucketsPerDocument[] spans = new SpansInBucketsPerDocument[2];
40  
41  	/** Do the Spans objects still point to valid hits? */
42  	private int currentDoc[] = new int[2];
43  
44  	/** The current hit (index into starts and ends lists) */
45  	private int currentHit = -1;
46  
47  	/** Starts of hits in current document */
48  	private IntArrayList starts = new IntArrayList();
49  
50  	/** Ends of hits in current document */
51  	private IntArrayList ends = new IntArrayList();
52  
53  	public SpansTags(Spans startTags, Spans endTags) {
54  		Spans[] origSpans = { startTags, endTags };
55  		for (int i = 0; i < 2; i++) {
56  			boolean sorted = false;
57  			if (origSpans[i] instanceof BLSpans)
58  				sorted = ((BLSpans) origSpans[i]).hitsStartPointSorted();
59  			else
60  				startTags = BLSpansWrapper.optWrap(origSpans[i]);
61  			if (sorted)
62  				spans[i] = new SpansInBucketsPerDocument(origSpans[i]);
63  			else
64  				spans[i] = new SpansInBucketsPerDocumentSorted(origSpans[i], cmpStartPoint);
65  		}
66  		currentDoc[0] = currentDoc[1] = -1;
67  	}
68  
69  	@Override
70  	public int docID() {
71  		return spans[0].docID();
72  	}
73  
74  	@Override
75  	public int startPosition() {
76  		if (currentHit < 0)
77  			return -1;
78  		if (currentHit >= starts.size())
79  			return NO_MORE_POSITIONS;
80  		return starts.get(currentHit);
81  	}
82  
83  	@Override
84  	public int endPosition() {
85  		if (currentHit < 0)
86  			return -1;
87  		if (currentHit >= starts.size())
88  			return NO_MORE_POSITIONS;
89  		return ends.get(currentHit);
90  	}
91  
92  	@Override
93  	public int nextDoc() throws IOException {
94  		// Hit list in current document exhausted. Are there more documents to do?
95  		if (currentDoc[0] == NO_MORE_DOCS || currentDoc[1] == NO_MORE_DOCS)
96  			return NO_MORE_DOCS;
97  
98  		// Move to the next document
99  		currentDoc[0] = spans[0].nextDoc();
100 		currentDoc[1] = spans[1].nextDoc();
101 		if (currentDoc[0] != currentDoc[1])
102 			throw new RuntimeException("Error, start and end tags not in synch");
103 		if (currentDoc[0] == NO_MORE_DOCS || currentDoc[1] == NO_MORE_DOCS)
104 			return NO_MORE_DOCS;
105 		gatherHits();
106 		return currentDoc[0];
107 	}
108 
109 	@Override
110 	public int nextStartPosition() throws IOException {
111 		// Do we have more hits in the list?
112 		currentHit++;
113 		if (currentHit < starts.size()) {
114 			// Yep, we're at the next hit in the list.
115 			return starts.get(currentHit);
116 		}
117 		return NO_MORE_POSITIONS;
118 	}
119 
120 	private void gatherHits() throws IOException {
121 		// Put the start and end tag positions in one list (ends negative)
122 		// (Note that we add 2 to the tag position to avoid the problem of x == -x for x == 0;
123 		//  below we subtract it again)
124 		// The list will be sorted by tag position.
125 		IntArrayList startsAndEnds = new IntArrayList();
126 		if (spans[0].nextBucket() == SpansInBuckets.NO_MORE_BUCKETS ||
127 			spans[1].nextBucket() == SpansInBuckets.NO_MORE_BUCKETS) {
128 			throw new RuntimeException("Both spans must have exactly one bucket per document");
129 		}
130 		int startIndex = 0, endIndex = 0;
131 		boolean startDone = false, endDone = false;
132 		int startBucketSize = spans[0].bucketSize();
133 		int endBucketSize = spans[1].bucketSize();
134 		while(!startDone || !endDone) {
135 			// Which comes first, the current start tag or end tag?
136 			int endTagPos = endDone ? -1 : spans[1].startPosition(endIndex);
137 			boolean addEndTag = false;
138 			int startTagPos = -1;
139 			if (startDone)
140 				addEndTag = true;
141 			else {
142 				startTagPos = spans[0].startPosition(startIndex);
143 				if (!endDone && startTagPos >= endTagPos)
144 					addEndTag = true;
145 			}
146 			if (addEndTag) {
147 				// Add end tag
148 				// +2 to avoid 0/-0 problem; -1 because endtag is attached to next token, but this
149 				// is inconvenient for this process, we want it attached to the previous token now.
150 				startsAndEnds.add(- (endTagPos + 2 - 1));
151 				endIndex++;
152 				if (endIndex >= endBucketSize)
153 					endDone = true;
154 			} else {
155 				// Add start tag
156 				startsAndEnds.add(startTagPos + 2);
157 				startIndex++;
158 				if (startIndex >= startBucketSize)
159 					startDone = true;
160 			}
161 		}
162 
163 		// Go through the list of all tags, keep track of unmatched open tags and finding matches
164 		IntArrayList unmatchedOpenTagIndices = new IntArrayList();
165 		IntArrayList emptyElementIndices = new IntArrayList(); // empty elements between tokens need special attention (see below)
166 
167 		// Don't always just call .clear() because the application could
168 		// keep holding on to too much memory after encountering one really
169 		// large document!
170 		if (starts.size() < ARRAYLIST_REALLOC_THRESHOLD) {
171 			// Not a huge amount of memory, so don't reallocate
172 			starts.clear();
173 			ends.clear();
174 		} else {
175 			// Reallocate in this case to avoid holding on to a lot of memory
176 			starts = new IntArrayList();
177 			ends = new IntArrayList();
178 		}
179 
180 		currentHit = -1; // before first hit
181 		IntIterator it = startsAndEnds.intIterator();
182 		while (it.hasNext()) {
183 			int tag = it.next();
184 			if (tag > 0) {
185 				int startPositionToStore = tag - 2;  // subtract 2 again to get original position (see above)
186 				if (!emptyElementIndices.isEmpty()) {
187 					// We have an unmatched close tag, probably an empty element between tokens (see below).
188 					// Is this the corresponding start tag?
189 					int index = emptyElementIndices.removeAtIndex(emptyElementIndices.size() - 1);
190 					if (startPositionToStore == ends.get(index)) {
191 						// Yes. Fill in start position.
192 						starts.set(index, startPositionToStore);
193 					} else {
194 						throw new RuntimeException("Unmatched close tag");
195 					}
196 				} else {
197 					// Open tag. As yet unmatched. Add to starts list, reserve space in ends list, add the index to the stack.
198 					starts.add(startPositionToStore);
199 					ends.add(-1); // "to be filled in later"
200 					unmatchedOpenTagIndices.add(ends.size() - 1); // index to fill in when end tag found
201 				}
202 			} else {
203 				// Close tag. Are there unmatched open tags?
204 				int endPositionToStore = -tag - 2 + 1; // subtract 2 and add 1 to let it point to the next token again (see above)
205 				if (unmatchedOpenTagIndices.isEmpty()) {
206 					// No. This must be an empty element between two tokens. Because we don't know the
207 					// relative order of tags between tokens (this information is not in the index),
208 					// our sort messes this up.
209 					// Add placeholder at this position and fix it when we find the open tag.
210 					starts.add(-1); // "to be filled in later"
211 					ends.add(endPositionToStore);
212 					emptyElementIndices.add(ends.size() - 1); // index to fill in when end tag found
213 				} else {
214 					// Yes. Match it to the most recently added unmatched open tag.
215 					int index = unmatchedOpenTagIndices.removeAtIndex(unmatchedOpenTagIndices.size() - 1);
216 					ends.set(index, endPositionToStore);
217 				}
218 			}
219 		}
220 		if (!unmatchedOpenTagIndices.isEmpty()) {
221 			throw new RuntimeException("Unmatched open tags left for document");
222 		}
223 		if (!emptyElementIndices.isEmpty()) {
224 			throw new RuntimeException("Unmatched close tag left for document");
225 		}
226 	}
227 
228 	/**
229 	 * Skip to the specified document (or the first document after it containing hits)
230 	 *
231 	 * @param doc
232 	 *            the doc number to skip to (or past)
233 	 * @return true if we're still pointing to a valid hit, false if we're done
234 	 * @throws IOException
235 	 */
236 	@Override
237 	public int advance(int doc) throws IOException {
238 		// Skip both
239 		currentDoc[0] = spans[0].advance(doc);
240 		currentDoc[1] = spans[1].advance(doc);
241 		if (currentDoc[0] != currentDoc[1])
242 			throw new RuntimeException("Error, start and end tags not in synch");
243 		if (currentDoc[0] == NO_MORE_DOCS || currentDoc[1] == NO_MORE_DOCS)
244 			return NO_MORE_DOCS;
245 		gatherHits();
246 		return currentDoc[0];
247 	}

249 	@Override
250 	public String toString() {
251 		return "SpansTags(" + spans[0] + ", " + spans[1] + ")";
252 	}

275 	public boolean hitsHaveUniqueStart() {
276 		// TODO: what happens if two of the same starttags occur at the same position..?
277 		// Right now, we assume this doesn't happen, and it usually doesn't, but if it
278 		// does, can it cause a mismatched tag problem?
279 		return true;
280 	}
