39  /**
40   * Translates a TextPattern to a Lucene SpanQuery object.
41   */
42  public class TextPatternTranslatorSpanQuery extends TextPatternTranslator<SpanQuery> {
43  
44  	@Override
45  	public SpanQuery and(QueryExecutionContext context, List<SpanQuery> clauses) {
46  		return new SpanQueryAnd(clauses);
47  	}
48  
49  	@Override
50  	public SpanQuery andNot(QueryExecutionContext context, SpanQuery include, SpanQuery exclude) {
51  		return new SpanQueryPositionFilter(include, exclude, TextPatternPositionFilter.Operation.MATCHES, true);
52  	}
53  
54  	@Override
55  	public SpanQuery or(QueryExecutionContext context, List<SpanQuery> clauses) {
56  		return new BLSpanOrQuery(clauses.toArray(new SpanQuery[] {}));
57  	}
58  
59  	@Override
60  	public SpanQuery regex(QueryExecutionContext context, String value) {
61  		String valueNoStartEndMatch = value.replaceAll("\\^|\\$", "");
62  		try {
63  			return new BLSpanMultiTermQueryWrapper<>(new RegexpQuery(
64  					new Term(context.luceneField(), context.subpropPrefix() + context.optDesensitize(valueNoStartEndMatch))));
65  		} catch (StackOverflowError e) {
66  			// If we pass in a really large regular expression, like a huge
67  			// list of words combined with OR, stack overflow occurs inside
68  			// Lucene's automaton building code and we may end up here.
69  			throw new RegexpTooLargeException();
70  		}
71  	}
72  
73  	@Override
74  	public SpanQuery sequence(QueryExecutionContext context, List<SpanQuery> clauses) {
75  		return new SpanQuerySequence(clauses);
76  	}
77  
78  	@Override
79  	public SpanQuery docLevelAnd(QueryExecutionContext context, List<SpanQuery> clauses) {
80  		return new SpanQueryDocLevelAnd(clauses);
81  	}
82  
83  	@Override
84  	public SpanQuery fuzzy(QueryExecutionContext context, String value, int maxEdits, int prefixLength) {
85  		String valuePrefix = context.subpropPrefix(); // for searching in "subproperties" (e.g. PoS features)
86  		prefixLength += valuePrefix.length();
87  		return new SpanFuzzyQuery(new Term(context.luceneField(), valuePrefix + context.optDesensitize(value)), maxEdits, prefixLength);
88  	}
89  
90  	@Override
91  	public SpanQuery tags(QueryExecutionContext context, String elementName, Map<String, String> attr) {
92  		SpanQuery allTags;
93  		if (context.tagLengthInPayload()) {
94  			// Modern index, with tag length in payload
95  			allTags = new SpanQueryTags(context, elementName);
96  		} else {
97  			// Older index, with end tags stored in separate property
98  			allTags = new SpanQueryTagsOld(context, elementName);
99  		}
100 		if (attr == null || attr.isEmpty())
101 			return allTags;
102 
103 		// Construct attribute filters
104 		List<SpanQuery> attrFilters = new ArrayList<>();
105 		QueryExecutionContext startTagContext = context.withProperty(ComplexFieldUtil.START_TAG_PROP_NAME);
106 		for (Map.Entry<String,String> e: attr.entrySet()) {
107 			String value = optInsensitive(context, "@" + e.getKey() + "__" + e.getValue());
108 			attrFilters.add(term(startTagContext, value));
109 		}
110 
111 		// Filter the tags
112 		// (NOTE: only works for start tags and full elements because attribute values
113 		//  are indexed at the start tag!)
114 		SpanQuery filter = new SpanQueryAnd(attrFilters);
115 		return new SpanQueryPositionFilter(allTags, filter, TextPatternPositionFilter.Operation.STARTS_AT, false);
116 	}
117 
118 	@Override
119 	public SpanQuery positionFilter(QueryExecutionContext context,
120 			SpanQuery producer, SpanQuery filter, Operation op, boolean invert,
121 			int leftAdjust, int rightAdjust) {
122 		return new SpanQueryPositionFilter(producer, filter, op, invert, leftAdjust, rightAdjust);
123 	}
124 
125 	@Override
126 	public SpanQuery startsAt(QueryExecutionContext context, SpanQuery producer, SpanQuery filter) {
127 		return new SpanQueryPositionFilter(producer, filter, TextPatternPositionFilter.Operation.STARTS_AT, false);
128 	}
129 
130 	@Override
131 	public SpanQuery endsAt(QueryExecutionContext context, SpanQuery producer, SpanQuery filter) {
132 		return new SpanQueryPositionFilter(producer, filter, TextPatternPositionFilter.Operation.ENDS_AT, false);
133 	}
134 
135 	@Override
136 	public SpanQuery term(QueryExecutionContext context, String value) {
137 		// Use a BlackLabSpanTermQuery instead of default Lucene one
138 		// because we need to override getField() to only return the base field name,
139 		// not the complete field name with the property.
140 		return new BLSpanTermQuery(new Term(context.luceneField(), context.subpropPrefix() + context.optDesensitize(value)));
141 	}
142 
143 	@Override
144 	public SpanQuery expand(QueryExecutionContext context, SpanQuery clause, boolean expandToLeft, int min, int max) {
145 		SpanQueryExpansion spanQueryExpansion = new SpanQueryExpansion(clause, expandToLeft, min, max);
146 		spanQueryExpansion.setIgnoreLastToken(context.alwaysHasClosingToken());
147 		return spanQueryExpansion;
148 	}
149 
150 	@Override
151 	public SpanQuery filterNGrams(QueryExecutionContext context, SpanQuery clause, Operation op, int min, int max) {
152 		return new SpanQueryFilterNGrams(clause, op, min, max);
153 	}
154 
155 	@Override
156 	public SpanQuery repetition(SpanQuery clause, int min, int max) {
157 		return new SpanQueryRepetition(clause, min, max);
158 	}
159 
160 	@Override
161 	public SpanQuery docLevelAndNot(SpanQuery include, SpanQuery exclude) {
162 		return new SpanQueryDocLevelAndNot(include, exclude);
163 	}
164 
165 	@Override
166 	public SpanQuery wildcard(QueryExecutionContext context, String value) {
167 		return new BLSpanMultiTermQueryWrapper<>(new WildcardQuery(new Term(context.luceneField(),
168 				context.subpropPrefix() + context.optDesensitize(value))));
169 	}
170 
171 	@Override
172 	public SpanQuery prefix(QueryExecutionContext context, String value) {
173 		return new BLSpanMultiTermQueryWrapper<>(new PrefixQuery(new Term(context.luceneField(),
174 				context.subpropPrefix() + context.optDesensitize(value))));
175 	}
176 
177 	@Override
178 	public SpanQuery not(QueryExecutionContext context, SpanQuery clause) {
179 		SpanQueryNot spanQueryNot = new SpanQueryNot(clause);
180 		spanQueryNot.setIgnoreLastToken(context.alwaysHasClosingToken());
181 		return spanQueryNot;
182 	}
183 
184 	@Override
185 	public SpanQuery any(QueryExecutionContext context, int min, int max) {
186 		return new SpanQueryNGrams(context.alwaysHasClosingToken(), context.luceneField(), min, max);
187 		//return SpanQueryNot.matchAllTokens(context.alwaysHasClosingToken(), context.luceneField());
188 	}
189 
190 	@Override
191 	public SpanQuery edge(SpanQuery clause, boolean rightEdge) {
192 		return new SpanQueryEdge(clause, rightEdge);
193 	}
194 
195 	@Override
196 	public SpanQuery captureGroup(SpanQuery clause, String name) {
197 		return new SpanQueryCaptureGroup(clause, name);
198 	}
