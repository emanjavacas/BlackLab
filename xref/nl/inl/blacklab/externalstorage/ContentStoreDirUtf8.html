39  /**
40   * Store string content by id in a directory of compound files with a TOC file. Quickly retrieve
41   * (parts of) the string content.
42   *
43   * This implementation stores the strings in UTF-8 encoding to save disk space. To guarantee
44   * reasonably fast random access times, we keep track of "block offsets", which are byte offsets to
45   * the start of (fixed char size) blocks. Block size in bytes can be slightly larger than char size
46   * because some UTF-8 characters take up more than 1 byte. If the block size is 1000 chars, block
47   * offsets might be [0, 1011, 2015, 3020].
48   */
49  public class ContentStoreDirUtf8 extends ContentStoreDirAbstract {
50  	/** Table of contents entry */
51  	static class TocEntry {
52  		/** id of the string */
53  		public int id;
54  
55  		/** id of the file the string was stored in */
56  		public int fileId;
57  
58  		/** byte offset into the file of the string */
59  		public int entryOffsetBytes;
60  
61  		/** length of the string in bytes */
62  		public int entryLengthBytes;
63  
64  		/** length of the string in characters */
65  		public int entryLengthCharacters;
66  
67  		/** fixed block size in characters (note that byte size differs per block) */
68  		public int blockSizeCharacters;
69  
70  		/** relative block start offsets in bytes */
71  		public int[] blockOffsetBytes;
72  
73  		/** was this entry deleted? (can be removed in next compacting run) */
74  		public boolean deleted;
75  
76  		public TocEntry(int id, int fileId, int offset, int length, int charLength, int blockSize,
77  				boolean deleted, int[] blockOffset) {
78  			super();
79  			this.id = id;
80  			this.fileId = fileId;
81  			entryOffsetBytes = offset;
82  			entryLengthBytes = length;
83  			entryLengthCharacters = charLength;
84  			blockSizeCharacters = blockSize;
85  			this.deleted = deleted;
86  			blockOffsetBytes = blockOffset;
87  		}

89  		/**
90  		 * Store TOC entry in the TOC file
91  		 *
92  		 * @param buf
93  		 *            where to serialize to
94  		 * @throws IOException
95  		 *             on error
96  		 */
97  		public void serialize(ByteBuffer buf) throws IOException {
98  			buf.putInt(id);
99  			buf.putInt(fileId);
100 			buf.putInt(entryOffsetBytes);
101 			buf.putInt(entryLengthBytes);
102 			buf.putInt(deleted ? -1 : entryLengthCharacters);
103 			buf.putInt(blockSizeCharacters);
104 			buf.putInt(blockOffsetBytes.length);
105 			IntBuffer ib = buf.asIntBuffer();
106 			ib.put(blockOffsetBytes);
107 			buf.position(buf.position() + blockOffsetBytes.length * Integer.SIZE / Byte.SIZE);
108 		}
109 
110 		/**
111 		 * Read TOC entry from the TOC file
112 		 *
113 		 * @param buf
114 		 *            the buffer to read from
115 		 * @return new TocEntry
116 		 * @throws IOException
117 		 *             on error
118 		 */
119 		public static TocEntry deserialize(ByteBuffer buf) throws IOException {
120 			int id = buf.getInt();
121 			int fileId = buf.getInt();
122 			int offset = buf.getInt();
123 			int length = buf.getInt();
124 			int charLength = buf.getInt();
125 			boolean deleted = charLength < 0;
126 			int blockSize = buf.getInt();
127 			int nBlocks = buf.getInt();
128 			int[] blockOffsetBytes = new int[nBlocks];
129 			IntBuffer ib = buf.asIntBuffer();
130 			ib.get(blockOffsetBytes);
131 			buf.position(buf.position() + blockOffsetBytes.length * Integer.SIZE / Byte.SIZE);
132 			return new TocEntry(id, fileId, offset, length, charLength, blockSize, deleted,
133 					blockOffsetBytes);
134 		}
135 
136 		/**
137 		 * Get the offset of the first byte of the specified block.
138 		 *
139 		 * @param blockNumber
140 		 *            which block?
141 		 * @return byte offset of the first byte in the file
142 		 */
143 		public long getBlockStartOffset(int blockNumber) {
144 			return entryOffsetBytes + blockOffsetBytes[blockNumber];
145 		}
146 
147 		/**
148 		 * Get the offset of the first byte beyond the specified block.
149 		 *
150 		 * @param blockNumber
151 		 *            which block?
152 		 * @return byte offset of the first byte beyond the block in the file
153 		 */
154 		public long getBlockEndOffset(int blockNumber) {
155 			if (blockNumber < blockOffsetBytes.length - 1)
156 				return entryOffsetBytes + blockOffsetBytes[blockNumber + 1];
157 			return entryOffsetBytes + entryLengthBytes;
158 		}
159 
160 		/**
161 		 * Size of this entry serialized
162 		 *
163 		 * @return the size in bytes
164 		 */
165 		public int sizeBytes() {
166 			return 28 + blockOffsetBytes.length * 4;
167 		}
168 	}

254 	/**
255 	 * What block size to use when adding a new document to the content store. Contributing factors
256 	 * for choosing block size:
257 	 * - larger blocks improve compression ratio
258 	 * - larger blocks decrease number of blocks you have to read
259 	 * - smaller blocks decrease the decompression time
260 	 * - smaller blocks increase the chance that we only have to read one disk block for a single concordance
261 	 *   (disk blocks are generally 2 or 4K)
262 	 * - consider OS I/O caching and memory mapping. Then it becomes the difference between reading
263 	 *   a few bytes from memory and reading a few kilobytes and decompressing them. Right now,
264 	 *   making concordances is often CPU-bound (because of decompression?)
265 	 */
266 	protected int newEntryBlockSizeCharacters = 4000;

289 	/**
290 	 * @param dir content store dir
291 	 * @param create if true, create a new content store
292 	 */
293 	public ContentStoreDirUtf8(File dir, boolean create) {
294 		this.dir = dir;
295 		if (!dir.exists())
296 			dir.mkdir();
297 		tocFile = new File(dir, "toc.dat");
298 		if (create && tocFile.exists()) {
299 			// Delete the ContentStore files
300 			tocFile.delete();
301 			new File(dir, "version.dat").delete();
302 			File[] dataFiles = dir.listFiles(new FilenameFilter() {
303 				@Override
304 				public boolean accept(File dir_, String name) {
305 					return name.matches("data\\d+.dat");
306 				}
307 			});
308 			for (File f : dataFiles) {
309 				f.delete();
310 			}
311 		}
312 		toc = Maps.mutable.empty();
313 		if (tocFile.exists())
314 			readToc();
315 		tocModified = false;
316 		if (create) {
317 			clear();
318 			if (tocFile.exists())
319 				tocFile.delete();
320 			setStoreType();
321 		}
322 		blockOffsetWhileStoring = new ArrayList<>();
323 	}

333 	/**
334 	 * Delete all content in the document store
335 	 */
336 	@Override
337 	public void clear() {
338 		closeCurrentStoreFile();
339 
340 		// delete all data files and empty TOC
341 		for (Map.Entry<Integer, TocEntry> me : toc.entrySet()) {
342 			TocEntry e = me.getValue();
343 			File f = getContentFile(e.fileId);
344 			if (f.exists())
345 				f.delete();
346 		}
347 		toc.clear();
348 		tocModified = true;
349 		currentFileId = 1;
350 		currentFileLength = 0;
351 		nextId = 1;
352 	}

354 	private void mapToc(boolean writeable) throws IOException {
355 		tocRaf = new RandomAccessFile(tocFile, writeable ? "rw" : "r");
356 		long fl = tocFile.length();
357 		if (writeable) {
358 			fl += writeMapReserve;
359 		} // leave 1M room at the end
360 		tocFileChannel = tocRaf.getChannel();
361 		tocFileBuffer = tocFileChannel.map(writeable ? MapMode.READ_WRITE : MapMode.READ_ONLY, 0,
362 				fl);
363 	}
364 
365 	private void closeMappedToc() {
366 		if (tocFileBuffer == null)
367 			return; // not mapped
368 		try {
369 			tocFileChannel.close();
370 			tocFileChannel = null;
371 			tocRaf.close();
372 			tocRaf = null;
373 
374 			tocFileBuffer = null;
375 
376 		} catch (IOException e) {
377 			throw new RuntimeException(e);
378 		}
379 	}

381 	/**
382 	 * Read the table of contents from the file
383 	 */
384 	private void readToc() {
385 		toc.clear();
386 		try {
387 			mapToc(false);
388 			try {
389 				tocFileBuffer.position(0);
390 				int n = tocFileBuffer.getInt();
391 				for (int i = 0; i < n; i++) {
392 					TocEntry e = TocEntry.deserialize(tocFileBuffer);
393 					toc.put(e.id, e);
394 
395 					// Keep track of the current content file (file with highest ID)
396 					if (e.fileId > currentFileId) {
397 						currentFileId = e.fileId;
398 						currentFileLength = 0;
399 					}
400 
401 					// Keep track of the length of the current content file
402 					if (e.fileId == currentFileId) {
403 						int endOfEntry = e.entryOffsetBytes + e.entryLengthBytes;
404 						if (endOfEntry > currentFileLength)
405 							currentFileLength = endOfEntry;
406 					}
407 
408 					// Keep track of what the next ID should be
409 					if (e.id + 1 > nextId)
410 						nextId = e.id + 1;
411 				}
412 			} finally {
413 				closeMappedToc();
414 			}
415 		} catch (Exception e) {
416 			throw ExUtil.wrapRuntimeException(e);
417 		}
418 	}
419 
420 	private void writeToc() {
421 		try {
422 			mapToc(true);
423 			tocFileBuffer.putInt(toc.size());
424 			try {
425 				for (TocEntry e : toc.values()) {
426 					if (tocFileBuffer.remaining() < e.sizeBytes()) {
427 						// Close and re-open with extra writing room
428 						int p = tocFileBuffer.position();
429 						closeMappedToc();
430 						mapToc(true);
431 						tocFileBuffer.position(p);
432 					}
433 					e.serialize(tocFileBuffer);
434 				}
435 			} finally {
436 				closeMappedToc();
437 			}
438 		} catch (Exception e) {
439 			throw ExUtil.wrapRuntimeException(e);
440 		}
441 		tocModified = false;
442 	}

475 	/**
476 	 * Encode and write the block we've compiled so far and reset for next block
477 	 * @param os where to write to block to
478 	 */
479 	public void writeCurrentBlock(OutputStream os) {
480 		try {
481 			// Encode and write block
482 
483 			// OPT: we could try memory-mapping the current content store file (not sure if
484 			// you can grow a file in size while memory-mapped? maybe just put it at a fixed
485 			// size?) for additional speed.
486 
487 			String blockContent = currentBlockContents.toString();
488 			if (blockContent.length() == 0)
489 				throw new RuntimeException("ERROR, tried to write an empty block");
490 			byte[] buf = encodeBlock(blockContent);
491 			os.write(buf);
492 			bytesWritten += buf.length;
493 			currentBlockContents = new StringBuilder(newEntryBlockSizeCharacters);
494 		} catch (IOException e) {
495 			throw new RuntimeException(e);
496 		}
497 	}
498 
499 	/**
500 	 * Store part of a piece of large content. This may be called several times to store chunks of
501 	 * content, but MUST be *finished* by calling the "normal" store() method. You may call store()
502 	 * with the empty string if you wish.
503 	 *
504 	 * @param content
505 	 *            the content to store
506 	 */
507 	@Override
508 	public synchronized void storePart(String content) {
509 		if (content.length() == 0)
510 			return;
511 		if (blockOffsetWhileStoring.size() == 0)
512 			blockOffsetWhileStoring.add(0); // first offset is always 0
513 
514 		// Calculate what charsFromEntryWritten will be after storing this part
515 		// (used to determine if we will cross a block boundary)
516 		int offsetAfterThisPart = charsFromEntryWritten + content.length();
517 
518 		// See if we're about to cross any block boundaries; if so, write the content up to the
519 		// first block boundary, save the new block offset, and repeat.
520 		int thisPartCharsWritten = 0, thisPartCharsLeftToWrite = content.length();
521 
522 		OutputStream os = openCurrentStoreFile();
523 		while (true) {
524 			// Will we cross a(nother) block boundary writing this part of the content?
525 			int nextBlockBoundary = blockOffsetWhileStoring.size() * newEntryBlockSizeCharacters;
526 			boolean willWeCrossBlockBoundary = (offsetAfterThisPart > nextBlockBoundary);
527 			if (!willWeCrossBlockBoundary) {
528 				// No; break out of this loop and write the last bit of content.
529 				break;
530 			}
531 
532 			// How many characters till the block boundary?
533 			int charsRemainingInBlock = nextBlockBoundary - charsFromEntryWritten;
534 
535 			// Are we at the block boundary already?
536 			if (charsRemainingInBlock > 0) {
537 				// No, not at the boundary yet; add last piece to block
538 				// and update bookkeeping variables
539 				addToBlock(content.substring(thisPartCharsWritten, thisPartCharsWritten
540 						+ charsRemainingInBlock));
541 				charsFromEntryWritten += charsRemainingInBlock;
542 
543 				thisPartCharsWritten += charsRemainingInBlock;
544 				thisPartCharsLeftToWrite -= charsRemainingInBlock;
545 			}
546 
547 			// We are now at a block boundary. Write the block and
548 			// store next block offset.
549 			if (currentBlockContents.length() > 0) {
550 				writeCurrentBlock(os);
551 				blockOffsetWhileStoring.add(bytesWritten);
552 			}
553 		}
554 		// No more block boundaries to cross. If there's any content left to write in the
555 		// current block,
556 		// do so now.
557 		if (thisPartCharsLeftToWrite > 0) {
558 			addToBlock(content.substring(thisPartCharsWritten, thisPartCharsWritten
559 					+ thisPartCharsLeftToWrite));
560 			charsFromEntryWritten += thisPartCharsLeftToWrite;
561 		}
562 	}

590 	/**
591 	 * Store the given content and assign an id to it
592 	 *
593 	 * @param content
594 	 *            the content to store
595 	 * @return the id assigned to the content
596 	 */
597 	@Override
598 	public synchronized int store(String content) {
599 		storePart(content);
600 		if (currentBlockContents.length() > 0) {
601 			// Write the last (not completely full) block
602 			OutputStream os = openCurrentStoreFile();
603 			writeCurrentBlock(os);
604 		}
605 		int[] blockOffsetArray = new int[blockOffsetWhileStoring.size()];
606 		int i = 0;
607 		for (Integer bo : blockOffsetWhileStoring) {
608 			blockOffsetArray[i] = bo;
609 			i++;
610 		}
611 		TocEntry e = new TocEntry(nextId, currentFileId, currentFileLength, bytesWritten,
612 				charsFromEntryWritten, newEntryBlockSizeCharacters, false, blockOffsetArray);
613 		nextId++;
614 		currentFileLength += bytesWritten;
615 		toc.put(e.id, e);
616 		tocModified = true;
617 		charsFromEntryWritten = 0;
618 		bytesWritten = 0;
619 		blockOffsetWhileStoring = new ArrayList<>();
620 		return e.id;
621 	}
622 
623 	private OutputStream openCurrentStoreFile() {
624 		try {
625 			boolean createNew = false;
626 			if (currentFileLength > dataFileSizeHint) {
627 				// Current file is full!
628 				currentFileId++;
629 				currentFileLength = 0;
630 				if (currentStoreFileStream != null)
631 					currentStoreFileStream.close();
632 				currentStoreFileStream = null;
633 				createNew = true; // start a new store file; delete if old stuff hanging around
634 			}
635 			if (currentStoreFileStream == null) {
636 				File f = getContentFile(currentFileId);
637 				if (createNew && f.exists())
638 					f.delete(); // leftover from previous index; delete
639 				currentStoreFileStream = new BufferedOutputStream(new FileOutputStream(f, true));
640 			}
641 			return currentStoreFileStream;
642 		} catch (Exception e) {
643 			throw ExUtil.wrapRuntimeException(e);
644 		}
645 	}
646 
647 	private void closeCurrentStoreFile() {
648 		try {
649 			if (currentStoreFileStream != null)
650 				currentStoreFileStream.close();
651 		} catch (IOException e) {
652 			throw new RuntimeException(e);
653 		}
654 	}

681 	/**
682 	 * Retrieve one or more substrings from the specified content.
683 	 *
684 	 * This is more efficient than retrieving the whole content, or retrieving parts in separate
685 	 * calls, because the file is only opened once and random access is used to read only the
686 	 * required parts.
687 	 *
688 	 * NOTE: if offset and length are both -1, retrieves the whole content. This is used by the
689 	 * retrieve(id) method.
690 	 *
691 	 * @param contentId
692 	 *            id of the entry to get substrings from
693 	 * @param start
694 	 *            the starting points of the substrings (in characters).
695 	 *            -1 means "start of document"
696 	 * @param end
697 	 *            the end points of the substrings (in characters).
698 	 *            -1 means "end of document"
699 	 * @return the parts
700 	 */
701 	@Override
702 	public synchronized String[] retrieveParts(int contentId, int[] start, int[] end) {
703 		try {
704 			// Find the correct TOC entry
705 			TocEntry e = toc.get(contentId);
706 			if (e == null || e.deleted)
707 				return null;
708 
709 			// Sanity-check parameters
710 			int n = start.length;
711 			if (n != end.length)
712 				throw new IllegalArgumentException("start and end must be of equal length");
713 
714 			// Create array for results
715 			String[] result = new String[n];
716 
717 			// Open the correct file
718 			try (FileInputStream fileInputStream = new FileInputStream(getContentFile(e.fileId))) {
719 				try (FileChannel fileChannel = fileInputStream.getChannel()) {
720 					// Retrieve the strings requested
721 					for (int i = 0; i < n; i++) {
722 						int a = start[i];
723 						int b = end[i];
724 
725 						if (a == -1)
726 							a = 0;
727 						if (b == -1)
728 							b = e.entryLengthCharacters;
729 
730 						// Check values
731 						if (a < 0 || b < 0) {
732 							throw new IllegalArgumentException("Illegal values, start = " + a + ", end = " + b);
733 						}
734 						if (a > e.entryLengthCharacters || b > e.entryLengthCharacters) {
735 							throw new IllegalArgumentException("Value(s) out of range, start = " + a
736 									+ ", end = " + b + ", content length = " + e.entryLengthCharacters);
737 						}
738 						if (b <= a) {
739 							throw new IllegalArgumentException(
740 									"Tried to read empty or negative length snippet (from " + a
741 											+ " to " + b + ")");
742 						}
743 
744 						// 1 - determine what blocks to read
745 						int firstBlock = a / e.blockSizeCharacters;
746 						int lastBlock = (b - 1) / e.blockSizeCharacters;
747 
748 						// 2 - read and decode blocks
749 						StringBuilder decoded = new StringBuilder();
750 						for (int j = firstBlock; j <= lastBlock; j++) {
751 							long readStartOffset = e.getBlockStartOffset(j);
752 							int bytesToRead = (int) (e.getBlockEndOffset(j) - readStartOffset);
753 							ByteBuffer buffer = ByteBuffer.allocate(bytesToRead);
754 							int bytesRead = fileChannel.read(buffer, readStartOffset);
755 							if (bytesRead < bytesToRead) {
756 								// Apparently, something went wrong.
757 								throw new RuntimeException("Not enough bytes read, " + bytesRead
758 										+ " < " + bytesToRead);
759 							}
760 							decoded.append(decodeBlock(buffer.array(), 0, bytesRead));
761 						}
762 
763 						// 3 - take just what we need
764 						int firstChar = a % e.blockSizeCharacters;
765 						result[i] = decoded.toString().substring(firstChar, firstChar + b - a);
766 					}
767 				}
768 			}
769 			return result;
770 		} catch (Exception e) {
771 			throw ExUtil.wrapRuntimeException(e);
772 		}
773 	}
