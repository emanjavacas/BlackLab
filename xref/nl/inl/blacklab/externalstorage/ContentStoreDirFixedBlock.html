48  /**
49   * Store string content by id in a compound file and a TOC file. Quickly retrieve
50   * (parts of) the string content.
51   *
52   * Stores files in a file containing fixed-length (4K) blocks of zipped UTF-8.
53   * A file allocation table keeps track of each file's blocks as well as the
54   * character offset associated with each block so we can quickly access the data.
55   * Free blocks will be re-used to save space.
56   */
57  public class ContentStoreDirFixedBlock extends ContentStoreDirAbstract {
58  	private static final Logger logger = Logger.getLogger(ContentStoreDirFixedBlock.class);
59  
60  	/** The type of content store. Written to version file and detected when opening. */
61  	private static final String CONTENT_STORE_TYPE_NAME = "fixedblock";
62  
63  	/** Version of this type of content store. Written to version file and detected when opening. */
64  	private static final String CURRENT_VERSION = "1";
65  
66  	/** Name of the version file */
67  	private static final String VERSION_FILE_NAME = "version.dat";
68  
69  	/** Name of the table of contents file */
70  	private static final String TOC_FILE_NAME = "toc.dat";
71  
72  	/** Name of the file containing all the original file contents (zipped) */
73  	private static final String CONTENTS_FILE_NAME = "file-contents.dat";
74  
75  	/** How many bytes an int consists of (used when repositioning file pointers) */
76  	private static final int BYTES_PER_INT = Integer.SIZE / Byte.SIZE;
77  
78  	/**
79  	 * Block size for the contents file.
80  	 *
81  	 * Contributing factors for choosing block size:
82  	 * - larger blocks improve compression ratio
83  	 * - larger blocks decrease number of blocks you have to read
84  	 * - smaller blocks decrease the decompression time
85  	 * - smaller blocks increase the chance that we only have to read one disk block for a single concordance
86  	 *   (disk blocks are generally 2 or 4K)
87  	 * - consider OS I/O caching and memory mapping. Then it becomes the difference between reading
88  	 *   a few bytes from memory and reading a few kilobytes and decompressing them. Right now,
89  	 *   making concordances is often CPU-bound (because of decompression?)
90  	 */
91  	private static final int BLOCK_SIZE_BYTES = 4096;
92  
93  	/** How small a block can get without triggering a retry with more input characters */
94  	private static final int MINIMUM_ACCEPTABLE_BLOCK_SIZE = BLOCK_SIZE_BYTES * 9 / 10;
95  
96  	/** The expected average compression factor */
97  	private static final float AVERAGE_COMPRESSION_FACTOR = 4;
98  
99  	/** A conservative estimate to avoid our estimates going over the block size */
100 	private static final float CONSERVATIVE_COMPRESSION_FACTOR = AVERAGE_COMPRESSION_FACTOR * 7 / 8;
101 
102 	/** How many characters we will usually be able to fit within a BLOCK_SIZE */
103 	private static final int TYPICAL_BLOCK_SIZE_CHARACTERS = (int)(BLOCK_SIZE_BYTES * CONSERVATIVE_COMPRESSION_FACTOR);
104 
105 	/** The expected maximum compression factor */
106 	private static final float MAX_COMPRESSION_FACTOR = 20;
107 
108 	/** Maximum byte size of unencoded block (we make the zip buffer one larger to detect when buffer space was insufficient) */
109 	private static final int MAX_BLOCK_SIZE_BYTES = (int)(BLOCK_SIZE_BYTES * MAX_COMPRESSION_FACTOR);
110 
111 	/** How many available characters will trigger a block write. */
112 	private static final int WRITE_BLOCK_WHEN_CHARACTERS_AVAILABLE = (int)(BLOCK_SIZE_BYTES * MAX_COMPRESSION_FACTOR);

114 	/** Table of contents entry */
115 	static class TocEntry {
116 
117 		/** content store id for this document */
118 		public int id;
119 
120 		/** length of the encoded string in bytes */
121 		public int entryLengthBytes;
122 
123 		/** length of the decoded string in characters */
124 		public int entryLengthCharacters;
125 
126 		/** blocks this document is stored in */
127 		public int[] blockIndices;
128 
129 		/** first character stored in each block */
130 		public int[] blockCharOffsets;
131 
132 		/** was this entry deleted? (can be removed in next compacting run) */
133 		public boolean deleted;
134 
135 		public TocEntry(int id, int length, int charLength, boolean deleted, int[] blockIndices, int[] blockCharOffsets) {
136 			super();
137 			this.id = id;
138 			entryLengthBytes = length;
139 			entryLengthCharacters = charLength;
140 			this.deleted = deleted;
141 			this.blockIndices = blockIndices;
142 			this.blockCharOffsets = blockCharOffsets;
143 		}
144 
145 		/**
146 		 * Store TOC entry in the TOC file
147 		 *
148 		 * @param buf
149 		 *            where to serialize to
150 		 * @throws IOException
151 		 *             on error
152 		 */
153 		public void serialize(ByteBuffer buf) throws IOException {
154 			buf.putInt(id);
155 			buf.putInt(entryLengthBytes);
156 			buf.putInt(deleted ? -1 : entryLengthCharacters);
157 			buf.putInt(blockIndices.length);
158 			IntBuffer ib = buf.asIntBuffer();
159 			ib.put(blockIndices);
160 			ib.put(blockCharOffsets);
161 			buf.position(buf.position() + blockIndices.length * BYTES_PER_INT * 2);
162 		}
163 
164 		/**
165 		 * Read TOC entry from the TOC file
166 		 *
167 		 * @param buf
168 		 *            the buffer to read from
169 		 * @return new TocEntry
170 		 * @throws IOException
171 		 *             on error
172 		 */
173 		public static TocEntry deserialize(ByteBuffer buf) throws IOException {
174 			int id = buf.getInt();
175 			int length = buf.getInt();
176 			int charLength = buf.getInt();
177 			boolean deleted = charLength < 0;
178 			int nBlocks = buf.getInt();
179 			int[] blockIndices = new int[nBlocks];
180 			int[] blockCharOffsets = new int[nBlocks];
181 			IntBuffer ib = buf.asIntBuffer();
182 			ib.get(blockIndices);
183 			ib.get(blockCharOffsets);
184 			buf.position(buf.position() + blockIndices.length * BYTES_PER_INT * 2);
185 			return new TocEntry(id, length, charLength, deleted, blockIndices, blockCharOffsets);
186 		}
187 
188 		/**
189 		 * Get the offset of the first byte of the specified block.
190 		 *
191 		 * @param blockNumber
192 		 *            which block?
193 		 * @return byte offset of the first byte in the file
194 		 */
195 		public int getBlockNumber(int blockNumber) {
196 			return blockIndices[blockNumber];
197 		}
198 
199 		/**
200 		 * Size of this entry serialized
201 		 *
202 		 * @return the size in bytes
203 		 */
204 		public int sizeBytes() {
205 			return 4 * BYTES_PER_INT + blockIndices.length * BYTES_PER_INT * 2;
206 		}
207 	}

296 	/**
297 	 * @param dir content store dir
298 	 * @param create if true, create a new content store
299 	 */
300 	public ContentStoreDirFixedBlock(File dir, boolean create) {
301 		this.dir = dir;
302 		if (!dir.exists())
303 			dir.mkdir();
304 		tocFile = new File(dir, TOC_FILE_NAME);
305 		contentsFile = new File(dir, CONTENTS_FILE_NAME);
306 		if (create && tocFile.exists()) {
307 			// Delete the ContentStore files
308 			tocFile.delete();
309 			new File(dir, VERSION_FILE_NAME).delete();
310 			new File(dir, CONTENTS_FILE_NAME).delete();
311 
312 			// Also delete old content store format files if present
313 			File[] dataFiles = dir.listFiles(new FilenameFilter() {
314 				@Override
315 				public boolean accept(File dir_, String name) {
316 					return name.matches("data\\d+.dat");
317 				}
318 			});
319 			for (File f : dataFiles) {
320 				f.delete();
321 			}
322 		}
323 		toc = IntObjectMaps.mutable.empty();  //Maps.mutable.empty();
324 		if (tocFile.exists())
325 			readToc();
326 		tocModified = false;
327 		if (create) {
328 			clear();
329 			if (tocFile.exists())
330 				tocFile.delete();
331 			setStoreType();
332 		}
333 		blockIndicesWhileStoring = new IntArrayList();
334 		blockCharOffsetsWhileStoring = new IntArrayList();
335 
336 		final int POOL_SIZE = 10;
337 		compresserPool = new SimpleResourcePool<Deflater>(POOL_SIZE){
338 			@Override
339 			public Deflater createResource() {
340 				return new Deflater();
341 			}
342 
343 			@Override
344 			public void destroyResource(Deflater resource) {
345 				resource.end();
346 			}
347 		};
348 		decompresserPool = new SimpleResourcePool<Inflater>(POOL_SIZE){
349 			@Override
350 			public Inflater createResource() {
351 				return new Inflater();
352 			}
353 
354 			@Override
355 			public void destroyResource(Inflater resource) {
356 				resource.end();
357 			}
358 		};
359 		zipbufPool = new SimpleResourcePool<byte[]>(POOL_SIZE){
360 			@Override
361 			public byte[] createResource() {
362 				return new byte[MAX_BLOCK_SIZE_BYTES+1]; // one larger to detect when buffer space was insufficient
363 			}
364 		};
365 	}

383 	private void mapToc(boolean writeable) throws IOException {
384 		tocRaf = new RandomAccessFile(tocFile, writeable ? "rw" : "r");
385 		long fl = tocFile.length();
386 		if (writeable) {
387 			fl += writeMapReserve;
388 		} // leave 1M room at the end
389 		tocFileChannel = tocRaf.getChannel();
390 		tocFileBuffer = tocFileChannel.map(writeable ? MapMode.READ_WRITE : MapMode.READ_ONLY, 0, fl);
391 	}
392 
393 	private void closeMappedToc() {
394 		if (tocFileBuffer == null)
395 			return; // not mapped
396 		try {
397 			tocFileChannel.close();
398 			tocFileChannel = null;
399 			tocRaf.close();
400 			tocRaf = null;
401 
402 			tocFileBuffer = null;
403 
404 		} catch (IOException e) {
405 			throw new RuntimeException(e);
406 		}
407 	}

409 	/**
410 	 * Read the table of contents from the file
411 	 */
412 	private void readToc() {
413 		toc.clear();
414 		try {
415 			mapToc(false);
416 			try {
417 				tocFileBuffer.position(0);
418 				int n = tocFileBuffer.getInt();
419 				totalBlocks = 0;
420 				for (int i = 0; i < n; i++) {
421 					TocEntry e = TocEntry.deserialize(tocFileBuffer);
422 					toc.put(e.id, e);
423 
424 					// Keep track of the number of blocks
425 					for (int bl: e.blockIndices) {
426 						if (bl > totalBlocks - 1)
427 							totalBlocks = bl + 1;
428 					}
429 
430 					// Keep track of what the next ID should be
431 					if (e.id + 1 > nextId)
432 						nextId = e.id + 1;
433 				}
434 			} finally {
435 				closeMappedToc();
436 			}
437 
438 			// Determine occupied blocks
439 			boolean[] blockOccupied = new boolean[totalBlocks]; // automatically initialized to false
440 			int numOccupied = 0;
441 			for (TocEntry e: toc) {
442 				for (int bl: e.blockIndices) {
443 					blockOccupied[bl] = true;
444 					numOccupied++;
445 				}
446 			}
447 			// Build the list of free blocks
448 			freeBlocks.clear();
449 			freeBlocks.ensureCapacity(totalBlocks - numOccupied);
450 			for (int i = 0; i < totalBlocks; i++) {
451 				if (!blockOccupied[i])
452 					freeBlocks.add(i);
453 			}
454 
455 		} catch (Exception e) {
456 			throw ExUtil.wrapRuntimeException(e);
457 		}
458 	}
459 
460 	private void writeToc() {
461 		try {
462 			mapToc(true);
463 			tocFileBuffer.putInt(toc.size());
464 			try {
465 				for (TocEntry e : toc.values()) {
466 					if (tocFileBuffer.remaining() < e.sizeBytes()) {
467 						// Close and re-open with extra writing room
468 						int p = tocFileBuffer.position();
469 						closeMappedToc();
470 						mapToc(true);
471 						tocFileBuffer.position(p);
472 					}
473 					e.serialize(tocFileBuffer);
474 				}
475 			} finally {
476 				closeMappedToc();
477 			}
478 		} catch (Exception e) {
479 			throw ExUtil.wrapRuntimeException(e);
480 		}
481 		tocModified = false;
482 	}

485 	 * Close the content store. Writes the table of contents (if modified)
486 	 */
487 	@Override
488 	public void close() {
489 		compresserPool.close();
490 		decompresserPool.close();
491 		zipbufPool.close();

496 		}

498 	}

500 	/**
501 	 * Encode and write the block we've compiled so far and reset for next block
502 	 * @param writeLastBlock if true, we'll write the last block too even if it's not full
503 	 */
504 	public void writeBlocks(boolean writeLastBlock) {
505 		ensureContentsFileOpen();
506 
507 		// Do we have a block to write?
508 		while (writeLastBlock && unwrittenContents.length() > 0 || unwrittenContents.length() >= WRITE_BLOCK_WHEN_CHARACTERS_AVAILABLE) {
509 			int lenBefore = unwrittenContents.length();
510 			byte[] encoded = encodeBlock(); // encode a number of characters to produce a 4K block
511 			int lenAfter = unwrittenContents.length();
512 			int charLen = lenBefore - lenAfter;
513 			int blockIndex = writeToFreeBlock(encoded);
514 			blockIndicesWhileStoring.add(blockIndex);
515 			blockCharOffsetsWhileStoring.add(charsFromEntryWritten);
516 			charsFromEntryWritten += charLen;
517 			bytesWritten += encoded.length;
518 		}
519 	}

521 	/**
522 	 * Writes the block data to a free block and returns the block number.
523 	 *
524 	 * @param encoded the block data
525 	 * @return the block number
526 	 */
527 	private int writeToFreeBlock(byte[] encoded) {
528 		int freeBlock;
529 		if (freeBlocks.size() == 0) {
530 			// Add a new one at the end
531 			totalBlocks++;
532 			freeBlock = totalBlocks - 1;
533 		} else {
534 			// Take the first from the list
535 			freeBlock = freeBlocks.removeAtIndex(0);
536 		}
537 		long offset = (long)freeBlock * BLOCK_SIZE_BYTES;
538 
539 		// Write data to the block
540 		try {
541 			fchContentsFile.position(offset);
542 			ByteBuffer buf = ByteBuffer.wrap(encoded);
543 			fchContentsFile.write(buf);
544 			// pad block with garbage
545 			buf = ByteBuffer.wrap(blockPadding, 0, BLOCK_SIZE_BYTES - encoded.length);
546 			fchContentsFile.write(buf);
547 			return freeBlock;
548 		} catch (IOException e) {
549 			throw new RuntimeException(e);
550 		}
551 	}

553 	/**
554 	 * Store part of a piece of large content. This may be called several times to store chunks of
555 	 * content, but MUST be *finished* by calling the "normal" store() method. You may call store()
556 	 * with the empty string if you wish.
557 	 *
558 	 * @param content
559 	 *            the content to store
560 	 */
561 	@Override
562 	public synchronized void storePart(String content) {
563 		if (content.length() == 0)
564 			return;
565 
566 		unwrittenContents.append(content);
567 		writeBlocks(false);
568 	}

570 	/**
571 	 * Store the given content and assign an id to it.
572 	 *
573 	 * @param content
574 	 *            the content to store
575 	 * @return the id assigned to the content
576 	 */
577 	@Override
578 	public synchronized int store(String content) {
579 		storePart(content);
580 		if (unwrittenContents.length() > 0) {
581 			// Write the last (not completely full) block
582 			writeBlocks(true);
583 		}
584 
585 		// Convert lists to arrays of primitives for storing
586 		int[] blockIndices = new int[blockIndicesWhileStoring.size()];
587 		int i = 0;
588 		IntIterator it = blockIndicesWhileStoring.intIterator();
589 		while (it.hasNext()) {
590 			blockIndices[i] = it.next();
591 			i++;
592 		}
593 		int[] blockCharOffsets = new int[blockCharOffsetsWhileStoring.size()];
594 		i = 0;
595 		it = blockCharOffsetsWhileStoring.intIterator();
596 		while (it.hasNext()) {
597 			blockCharOffsets[i] = it.next();
598 			i++;
599 		}
600 
601 		TocEntry e = new TocEntry(nextId, bytesWritten, charsFromEntryWritten, false, blockIndices, blockCharOffsets);
602 		nextId++;
603 		toc.put(e.id, e);
604 		tocModified = true;
605 		charsFromEntryWritten = 0;
606 		bytesWritten = 0;
607 		blockIndicesWhileStoring.clear();
608 		blockCharOffsetsWhileStoring.clear();
609 		return e.id;
610 	}

612 	private void ensureContentsFileOpen() {
613 		try {
614 			if (rafContentsFile == null) {
615 				File theContentsFile = new File(dir, CONTENTS_FILE_NAME);
616 				rafContentsFile = new RandomAccessFile(theContentsFile, "rw");
617 				fchContentsFile = rafContentsFile.getChannel();
618 			}
619 		} catch (FileNotFoundException e) {
620 			throw new RuntimeException("Contents file not found" + CONTENTS_FILE_NAME, e);
621 		}
622 	}

624 	private void closeContentsFile() {
625 		try {
626 			if (rafContentsFile != null) {
627 				fchContentsFile.close();
628 				fchContentsFile = null;
629 				rafContentsFile.close();
630 				rafContentsFile = null;
631 			}
632 		} catch (IOException e) {
633 			throw new RuntimeException(e);
634 		}
635 	}

650 	/**
651 	 * Retrieve one or more substrings from the specified content.
652 	 *
653 	 * This is more efficient than retrieving the whole content, or retrieving parts in separate
654 	 * calls, because the file is only opened once and random access is used to read only the
655 	 * required parts.
656 	 *
657 	 * NOTE: if offset and length are both -1, retrieves the whole content. This is used by the
658 	 * retrieve(id) method.
659 	 *
660 	 * @param contentId
661 	 *            id of the entry to get substrings from
662 	 * @param start
663 	 *            the starting points of the substrings (in characters).
664 	 *            -1 means "start of document"
665 	 * @param end
666 	 *            the end points of the substrings (in characters).
667 	 *            -1 means "end of document"
668 	 * @return the parts
669 	 */
670 	@Override
671 	public synchronized String[] retrieveParts(int contentId, int[] start, int[] end) {
672 		try {
673 			// Find the correct TOC entry
674 			TocEntry e = toc.get(contentId);
675 			if (e == null || e.deleted)
676 				return null;
677 
678 			// Sanity-check parameters
679 			int n = start.length;
680 			if (n != end.length)
681 				throw new IllegalArgumentException("start and end must be of equal length");
682 
683 			// Create array for results
684 			String[] result = new String[n];
685 
686 			// Open the file
687 			try (FileInputStream fileInputStream = new FileInputStream(contentsFile)) {
688 				try (FileChannel fileChannel = fileInputStream.getChannel()) {
689 					// Retrieve the strings requested
690 					for (int i = 0; i < n; i++) {
691 						int a = start[i];
692 						int b = end[i];
693 
694 						if (a == -1)
695 							a = 0;
696 						if (b == -1)
697 							b = e.entryLengthCharacters;
698 
699 						// Check values
700 						if (a < 0 || b < 0) {
701 							throw new IllegalArgumentException("Illegal values, start = " + a + ", end = " + b);
702 						}
703 						if (a > e.entryLengthCharacters || b > e.entryLengthCharacters) {
704 							throw new IllegalArgumentException("Value(s) out of range, start = " + a
705 									+ ", end = " + b + ", content length = " + e.entryLengthCharacters);
706 						}
707 						if (b <= a) {
708 							throw new IllegalArgumentException(
709 									"Tried to read empty or negative length snippet (from " + a
710 											+ " to " + b + ")");
711 						}
712 
713 						// 1 - determine what blocks to read
714 						int firstBlock = -1, lastBlock = -1;
715 						int bl = 0;
716 						int charOffset = -1;
717 						for (int offs: e.blockCharOffsets) {
718 							if (offs <= a) {
719 								firstBlock = bl; // last block that starts before a
720 								charOffset = offs;
721 							}
722 							if (offs > b && lastBlock == -1) {
723 								lastBlock = bl - 1;  // first block that ends after b
724 								break;
725 							}
726 							bl++;
727 						}
728 						if (lastBlock == -1)
729 							lastBlock = bl - 1; // last available block
730 
731 						// 2 - read and decode blocks
732 						StringBuilder decoded = new StringBuilder();
733 						for (int j = firstBlock; j <= lastBlock; j++) {
734 							long blockNum = e.getBlockNumber(j);
735 							long readStartOffset = blockNum * BLOCK_SIZE_BYTES;
736 							int bytesToRead = BLOCK_SIZE_BYTES;
737 							ByteBuffer buffer = ByteBuffer.allocate(bytesToRead);
738 							int bytesRead = fileChannel.read(buffer, readStartOffset);
739 							if (bytesRead < bytesToRead) {
740 								// Apparently, something went wrong.
741 								throw new RuntimeException("Not enough bytes read, " + bytesRead
742 										+ " < " + bytesToRead);
743 							}
744 							String decodedBlock = decodeBlock(buffer.array(), 0, bytesRead);
745 							decoded.append(decodedBlock);
746 						}
747 
748 						// 3 - take just what we need
749 						int firstChar = a - charOffset;
750 						result[i] = decoded.substring(firstChar, firstChar + b - a);
751 					}
752 				}
753 			}
754 			return result;
755 		} catch (Exception e) {
756 			throw ExUtil.wrapRuntimeException(e);
757 		}
758 	}

796 	protected byte[] encodeBlock() {
797 
798 		int length = TYPICAL_BLOCK_SIZE_CHARACTERS;
799 		int available = unwrittenContents.length();
800 		if (length > available)
801 			length = available;
802 
803 		Deflater compresser = compresserPool.acquire();
804 		byte[] zipbuf = zipbufPool.acquire();
805 		boolean doMinCheck = true;
806 		try {
807 			while (true) {
808 
809 				// Serialize to bytes
810 				byte[] encoded;
811 				while (true) {
812 					encoded = unwrittenContents.substring(0, length).getBytes(DEFAULT_CHARSET);
813 
814 					// Make sure the block fits in our zip buffer
815 					if (encoded.length <= MAX_BLOCK_SIZE_BYTES)
816 						break;
817 					length -= (encoded.length - MAX_BLOCK_SIZE_BYTES) * 2;
818 					doMinCheck = false;
819 				}
820 
821 				// Compress
822 				compresser.reset();
823 				compresser.setInput(encoded);
824 				compresser.finish();
825 				int compressedDataLength = compresser.deflate(zipbuf, 0, zipbuf.length, Deflater.FULL_FLUSH);
826 				if (compressedDataLength <= 0) {
827 					throw new RuntimeException("Error, deflate returned " + compressedDataLength);
828 				}
829 				if (compressedDataLength == zipbuf.length) {
830 					throw new RuntimeException("Error, deflate returned size of zipbuf, this indicates insufficient space");
831 				}
832 
833 				// Check the size
834 				float waste = (float)(BLOCK_SIZE_BYTES - compressedDataLength) / BLOCK_SIZE_BYTES;
835 				float ratio = (float)length / compressedDataLength;
836 
837 				if (compressedDataLength > BLOCK_SIZE_BYTES) {
838 					// Compressed block too large.
839 					// Shrink the uncompressed data length by 5% more than what we expect to be required.
840 					float shrinkFactor = 1.0f + (1.05f * (compressedDataLength - BLOCK_SIZE_BYTES)) / BLOCK_SIZE_BYTES;
841 					logger.debug("Block size too large, retrying. Char length: " + length + ", encoded length: " + compressedDataLength + " > " + BLOCK_SIZE_BYTES + ", shrinkFactor: " + shrinkFactor);
842 					length = (int)(length / shrinkFactor);
843 					if (length <= 0)
844 						length = 1;
845 					doMinCheck = false; // prevent oscillation between enlarging and shrinking
846 				} else if (doMinCheck && length < available && compressedDataLength < MINIMUM_ACCEPTABLE_BLOCK_SIZE) {
847 					// Compressed block too small.
848 					// Grow the uncompressed data length by 5% less than what we expect is possible.
849 					float growFactor = 1.0f + (0.95f * (BLOCK_SIZE_BYTES - compressedDataLength)) / compressedDataLength;
850 					logger.debug("Block size too small, retrying. Char length: " + length + ", encoded length: " + compressedDataLength + " < " + MINIMUM_ACCEPTABLE_BLOCK_SIZE + ", growFactor: " + growFactor);
851 					length = (int)(length * growFactor);
852 					if (length > available)
853 						length = available;
854 				} else {
855 					logger.debug("Block ok. Char length: " + length + ", encoded length: " + compressedDataLength + ", waste%: " + waste + ", ratio: " + ratio);
856 					unwrittenContents.delete(0, length);
857 					return Arrays.copyOfRange(zipbuf, 0, compressedDataLength);
858 				}
859 			}
860 		} finally {
861 			compresserPool.release(compresser);
862 			zipbufPool.release(zipbuf);
863 		}
864 	}
865 
866 	protected String decodeBlock(byte[] buf, int offset, int length) {
867 		try {
868 			// unzip block
869 			Inflater decompresser = decompresserPool.acquire();
870 			byte[] zipbuf = zipbufPool.acquire();
871 			try {
872 				decompresser.reset();
873 				decompresser.setInput(buf, offset, length);
874 				int resultLength = decompresser.inflate(zipbuf);
875 				if (resultLength <= 0) {
876 					throw new RuntimeException("Error, inflate returned " + resultLength);
877 				}
878 				if (!decompresser.finished()) {
879 					// This shouldn't happen because our max block size prevents it
880 					throw new RuntimeException("Unzip buffer size insufficient");
881 				}
882 				return new String(zipbuf, 0, resultLength, DEFAULT_CHARSET);
883 			} finally {
884 				decompresserPool.release(decompresser);
885 				zipbufPool.release(zipbuf);
886 			}
887 		} catch (DataFormatException e) {
888 			throw new RuntimeException(e);
889 		}
890 	}
