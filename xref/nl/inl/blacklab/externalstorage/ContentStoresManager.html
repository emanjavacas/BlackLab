1   package nl.inl.blacklab.externalstorage;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   import org.apache.lucene.document.Document;
7   
8   import nl.inl.blacklab.search.ContentAccessor;
9   
10  public class ContentStoresManager {
11  	/**
12  	 * ContentAccessors tell us how to get a field's content:
13  	 * <ol>
14  	 * <li>if there is no contentaccessor: get it from the Lucene index (stored field)</li>
15  	 * <li>from an external source (file, database) if it's not (because the content is very large
16  	 * and/or we want faster random access to the content than a stored field can provide)</li>
17  	 * </ol>
18  	 *
19  	 * Indexed by complex field name.
20  	 */
21  	private Map<String, ContentAccessor> contentAccessors = new HashMap<>();
22  
23  	public void close() {
24  		// Close the content accessor(s)
25  		// (the ContentStore, and possibly other content accessors
26  		// (although that feature is not used right now))
27  		for (ContentAccessor ca: contentAccessors.values()) {
28  			ca.close();
29  		}
30  
31  	}
32  
33  	public void put(String field, ContentStore store) {
34  		contentAccessors.put(field, new ContentAccessor(field, store));
35  	}
36  
37  	public ContentStore get(String fieldName) {
38  		ContentAccessor ca = contentAccessors.get(fieldName);
39  		if (ca == null)
40  			return null;
41  		return ca.getContentStore();
42  	}
43  
44  	public void deleteDocument(Document d) {
45  		for (ContentAccessor ca: contentAccessors.values()) {
46  			ca.delete(d);
47  		}
48  	}
49  
50  	public boolean exists(String fieldName) {
51  		return contentAccessors.containsKey(fieldName);
52  	}
53  
54  	public String[] getSubstrings(String fieldName, Document d, int[] start, int[] end) {
55  		ContentAccessor contentAccessor = contentAccessors.get(fieldName);
56  		if (contentAccessor == null)
57  			return null;
58  		return contentAccessor.getSubstringsFromDocument(d, start, end);
59  	}
60  
61  	public String[] getSubstrings(String fieldName, int contentId, int[] start, int[] end) {
62  		ContentAccessor contentAccessor = contentAccessors.get(fieldName);
63  		if (contentAccessor == null)
64  			return null;
65  		return contentAccessor.getSubstringsFromDocument(contentId, start, end);
66  	}
67  
68  
69  }
