57  			usage();
58  			System.exit(1);
59  		}
60  
61  		Term term = new Term(fieldName, word);
62  		System.out.println("TERM: " + term.text());
63  
64  		System.out.println("  Total frequency: " + term.text() + ": " + reader.totalTermFreq(term) + "\n");
65  
66  		doQuery(term, reader);
67  		doSpanQuery(term, reader);
68  		if (matchingDoc != -1)
69  			doTermVector(matchingDoc, term, reader);
70  
71  		if (matchPosition != -1)
72  			doSnippet(matchingDoc, matchPosition, term, reader);

75  	private static void doSnippet(int docId, int position, Term term,
76  			IndexReader reader) {
77  
78  		System.out.println("\nSNIPPET");
79  
80  		int first = position - 10;
81  		if (first < 0)
82  			first = 0;
83  		int number = 20;
84  		String[] words = LuceneUtil.getWordsFromTermVector(reader, docId, term.field(), first, first + number, true);
85  
86  		// Print result
87  		int i = 0;
88  		StringBuilder b = new StringBuilder();
89  		for (String word: words) {
90  			if (word == null)
91  				word = "[MISSING]";
92  			b.append(i + first).append(":").append(word).append(" ");
93  			i++;
94  		}
95  		System.out.println(StringUtil.wrap(b.toString(), 80));
96  	}

98  	private static void doTermVector(int doc, Term term, IndexReader reader) {
99  		System.out.println("\nTERM VECTOR FOR DOC " + doc);
100 		String luceneName = term.field();
101 		String word = term.text();
102 		try {
103 			org.apache.lucene.index.Terms terms = reader.getTermVector(doc, luceneName);
104 			if (terms == null) {
105 				System.out.println("Field " + luceneName + " has no Terms");
106 				return;
107 			}
108 			System.out.println(
109 				"  Doc count:           " + terms.getDocCount() + "\n" +
110 				"  Sum doc freq:        " + terms.getSumDocFreq() + "\n" +
111 				"  Sum total term freq: " + terms.getSumDocFreq() + "\n" +
112 				"  Has offsets:         " + terms.hasOffsets() + "\n" +
113 				"  Has payloads:        " + terms.hasPayloads() + "\n" +
114 				"  Has positions:       " + terms.hasPositions() + "\n");
115 
116 			TermsEnum termsEnum = terms.iterator();
117 			if (!termsEnum.seekExact(term.bytes())) {
118 				System.out.println("Term " + word + " not found.");
119 				return;
120 			}
121 
122 			System.out.println("\n" +
123 				"  TERM '" + termsEnum.term().utf8ToString() + "':\n" +
124 				"    Doc freq:        " + termsEnum.docFreq() + "\n" +
125 				"    Ord:             " + termsEnum.ord() + "\n" +
126 				"    Total term freq: " + termsEnum.totalTermFreq() + "\n");
127 
128 			if (terms.hasPositions()) {
129 				// Verzamel concordantiewoorden uit term vector
130 				PostingsEnum docPosEnum = termsEnum.postings(null, null, PostingsEnum.POSITIONS);
131 				System.out.println("    POSITIONS:");
132 				while (docPosEnum.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
133 
134 					System.out.println(
135 							"      Doc id: " + docPosEnum.docID() + "\n" +
136 							"      Freq:   " + docPosEnum.freq() + "\n");
137 
138 					for (int i = 0; i < docPosEnum.freq(); i++)  {
139 						int position = docPosEnum.nextPosition();
140 						int start = docPosEnum.startOffset();
141 						if (start >= 0) {
142 							// Character offsets stored
143 							int end = docPosEnum.startOffset();
144 							System.out.println("      " + position + " (offsets: " + start + "-" + end + ")");
145 						} else {
146 							// No character offsets stored
147 							System.out.println("      " + position);
148 						}
149 						if (matchPosition < 0)
150 							matchPosition = position;
151 					}
152 				}
153 			} else {
154 				// No positions
155 				PostingsEnum postingsEnum = termsEnum.postings(null, null, PostingsEnum.FREQS);
156 				System.out.println("\n    DOCS:");
157 				while (postingsEnum.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
158 					System.out.println(
159 							"      Doc id: " + postingsEnum.docID() + "\n" +
160 							"      Freq:   " + postingsEnum.freq() + "\n");
161 				}
162 			}
163 		} catch (Exception e) {
164 			throw new RuntimeException(e);
165 		}
166 	}

168 	private static void doQuery(Term term, IndexReader reader) throws IOException {
169 		Query query = new TermQuery(term);
170 		query = query.rewrite(reader);
171 		System.out.println("REGULAR QUERY");
172 
173 		IndexSearcher searcher = new IndexSearcher(reader);
174 		final BitSet bits = new BitSet(reader.maxDoc());
175 		docsFound = false;
176 		searcher.search(query, new SimpleCollector() {
177 			private int docBase;
178 
179 			@Override
180 			public void setScorer(Scorer scorer) {
181 				// ignore scorer
182 			}
183 
184 			@Override
185 			protected void doSetNextReader(LeafReaderContext context)
186 					throws IOException {
187 				docBase = context.docBase;
188 				super.doSetNextReader(context);
189 			}
190 
191 			@Override
192 			public void collect(int doc) {
193 				bits.set(doc + docBase);
194 				System.out.println(String.format("  doc %7d", doc + docBase));
195 				matchingDoc = doc + docBase;
196 				docsFound = true;
197 			}
198 
199 			@Override
200 			public boolean needsScores() {
201 				return false;
202 			}
203 		});
204 		if (!docsFound)
205 			System.out.println("  (no matching docs)");
206 		System.out.println("");
207 	}

209 	private static void doSpanQuery(Term term, IndexReader reader) throws IOException {
210 		SpanQuery spanQuery = new SpanTermQuery(term);
211 		spanQuery = (SpanQuery) spanQuery.rewrite(reader);
212 
213 		Map<Term, TermContext> termContexts = new HashMap<>();
214 		TreeSet<Term> terms = new TreeSet<>();
215 		extractTermsFromSpanQuery(spanQuery, terms);
216 		for (Term termForContext: terms) {
217 			termContexts.put(termForContext, TermContext.build(reader.getContext(), termForContext));
218 		}
219 
220 		System.out.println("SPANQUERY");
221 
222 		System.out.println("USING LEAVES:");
223 		boolean hitsFound = false;
224 		for (LeafReaderContext arc: reader.leaves()) {
225 			Spans spans = spanQuery.getSpans(arc, arc.reader().getLiveDocs(), termContexts);
226 			while(spans != null && spans.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
227 				while (spans.nextStartPosition() != Spans.NO_MORE_POSITIONS) {
228 					int doc = arc.docBase + spans.docID();
229 					System.out.println(String.format("  doc %7d, pos %4d-%4d", doc, spans.startPosition(), spans.endPosition()));
230 					hitsFound = true;
231 				}
232 			}
233 		}
234 		if (!hitsFound)
235 			System.out.println("  (no hits)");
236 		System.out.println("");
237 
238 		System.out.println("USING SLOWCOMPOSITEREADERWRAPPER:");
239 		LeafReader scrw = SlowCompositeReaderWrapper.wrap(reader);
240 		Spans spans = spanQuery.getSpans(scrw.getContext(), scrw.getLiveDocs(), termContexts);
241 		hitsFound = false;
242 		while(spans != null && spans.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
243 			while(spans.nextStartPosition() != Spans.NO_MORE_POSITIONS) {
244 				System.out.println(String.format("  doc %7d, pos %4d-%4d", spans.docID(), spans.startPosition(), spans.endPosition()));
245 				hitsFound = true;
246 			}
247 		}
248 		if (!hitsFound)
249 			System.out.println("  (no hits)");
250 		System.out.println("");
251 	}
