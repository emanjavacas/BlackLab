38  /**
39   * Convert the ANW Corpus Sketch format files to almost-identical XML files. Using XML is useful
40   * because it allows us to later easily display it using XSLT.
41   */
42  public class SketchToXmlConverter {
43  	public static void main(String[] args) throws Exception {
44  		// Read property file
45  		Properties properties = getPropertiesFromResource("anwcorpus.properties");
46  
47  		File inDir = getFileProp(properties, "sketchDir", null);
48  		File listFile = new File(inDir, "lijst.txt");
49  
50  		File outDir = getFileProp(properties, "inputDir", "input", null);
51  
52  		SketchToXmlConverter.convertList(listFile, inDir, outDir);
53  	}

55  	/**
56  	 * Read Properties from a resource
57  	 *
58  	 * @param resourcePath
59  	 *            file path, relative to the classpath, where the properties file is
60  	 * @return the Properties read
61  	 * @throws IOException
62  	 */
63  	public static Properties getPropertiesFromResource(String resourcePath) throws IOException {
64  		Properties properties;
65  		InputStream isProps = SketchToXmlConverter.class.getClassLoader().getResourceAsStream(resourcePath);
66  		if (isProps == null) {
67  			// TODO: FileNotFoundException?
68  			throw new IllegalArgumentException("Properties file not found: " + resourcePath
69  					+ " (must be accessible from the classpath)");
70  		}
71  		try {
72  			properties = new Properties();
73  			properties.load(isProps);
74  			return properties;
75  		} finally {
76  			isProps.close();
77  		}
78  	}

98  	/**
99  	 * Get a File property from a Properties object.
100 	 *
101 	 * This may be an absolute file path (starts with / or \ or a Windows drive letter spec), or a
102 	 * path relative to basePath
103 	 *
104 	 * @param properties
105 	 *            where to read the value from
106 	 * @param name
107 	 *            the value's name
108 	 * @param defaultValue default value if the property was not specified
109 	 * @param basePath
110 	 *            base path the file path may be relative to
111 	 * @return the file, or null if not found
112 	 */
113 	public static File getFileProp(Properties properties, String name, String defaultValue, File basePath) {
114 		Object prop = properties.get(name);
115 		if (prop == null)
116 			prop = defaultValue;
117 		if (prop == null)
118 			return null;
119 		File filePath = new File(prop.toString());
120 
121 		// Is it an absolute path, or no base path given?
122 		if (basePath == null || filePath.isAbsolute()) {
123 			// Yes; ignore our base directory
124 			return filePath;
125 		}
126 		// Relative path; concatenate with base directory
127 		return new File(basePath, filePath.getPath());
128 	}
129 	private static final int LINES_PER_CHUNK_FILE = 30000;
130 
131 	boolean inSentence = false;
132 
133 	boolean inDoc = false;
134 
135 	private int linesDone;
136 
137 	private boolean docWasEmpty = true;
138 
139 	private String lastDocLine = "(no docs processed yet)";
140 
141 	public boolean processLine(String line, Writer out) throws IOException {
142 		boolean shouldContinue = true;
143 		if (line.length() >= 1 && line.charAt(0) == '<' && line.endsWith(">")) // tag?
144 		{
145 			if (line.charAt(1) == '/') {
146 				// end tag
147 				if (line.equals("</s>")) {
148 					if (!inSentence)
149 						System.err.println("Sentence close without open!");
150 					inSentence = false;
151 					docWasEmpty = false;
152 				} else if (line.equals("</doc>")) {
153 					if (!inDoc)
154 						System.err.println("Doc close without open! After: " + lastDocLine);
155 					if (docWasEmpty)
156 						System.err.println("Empty document");
157 					if (inSentence)
158 						out.append("</s>\n"); // close last sentence tag!
159 					inSentence = false;
160 					inDoc = false;
161 					if (linesDone >= LINES_PER_CHUNK_FILE) {
162 						shouldContinue = false;
163 					}
164 				} else
165 					System.err.println("Unknown end tag: " + line);
166 			} else {
167 				if (line.equals("<s>")) {
168 					if (inSentence)
169 						System.err.println("Nested sentence!");
170 					docWasEmpty = false;
171 					inSentence = true;
172 				} else if (line.equals("<g/>")) {
173 					// do nothing
174 				} else if (line.startsWith("<doc")) {
175 					lastDocLine = line;
176 					if (inDoc) {
177 						out.append("</doc>\n"); // close unclosed doc
178 						System.err.println("--- Unclosed " + (docWasEmpty ? "empty " : "")
179 								+ "document before: " + line);
180 						System.err.println("    Fixed: (added close tag)");
181 					}
182 
183 					// Fix o.a. subcorpus="Domeinen" id="7637"
184 					// [v2]
185 					// line = line.replaceAll("\\s+jaar=\"(\\d+)\"\"\\s+", " jaar=\"$1\" ");
186 
187 					// Quotes om jaar
188 					// [v2]
189 					// line = line.replaceAll("\\s+jaar=(\\d+)\\s+", " jaar=\"$1\" ");
190 
191 					// Properly escape &'s
192 					line = line.replaceAll("&", "&amp;");
193 
194 					// Spaties normaliseren
195 					line = line.replaceAll("\\s\\s+", " ");
196 
197 					if (!line.matches("<doc(\\s+\\w+\\s*=\\s*\"[^<>\"]*\")*\\s*>\\s*")) {
198 						System.err.println("--- Illegal doc line: " + line);
199 
200 						// Fix subcorpus="Pluscorpus" id="68795"
201 						line = line.replaceAll("<URL >", "url=\"");
202 						line = line.replaceAll("/URL</URL>", "\"");
203 
204 						// Fix subcorpus="CLT" id="69591"
205 						line = line
206 								.replaceAll("<CLTSTRATUM>EON</CLTSTRATUM>", "cltstratum=\"EON\"");
207 
208 						// Fix subcorpus="Domeinen" id="2312"
209 						line = line.replaceAll("auteur=\"\" ", "auteur=\"");
210 
211 						// [v2]
212 						line = line.replaceAll("auteurwebtekst=\"\" ", "auteurwebtekst=\"");
213 
214 						// Fix subcorpus="Domeinen" id="69831"
215 						line = line.replaceAll("<FILEDESC\\s+>DOMEIN\\s+</FILEDESC>",
216 								"filedesc=\"DOMEIN\"");
217 
218 						// Fix subcorpus="CLT" id="478"
219 						line = line.replaceAll("<TEKST>", "tekst=\"");
220 						line = line.replaceAll("</TEKST>", "\"");
221 
222 						// Check that that fixed it
223 						if (!line.matches("<doc(\\s+\\w+\\s*=\\s*\"[^<>\"]*\")*\\s*>\\s*")) {
224 							System.err.println("!!! Failed: " + line);
225 						} else {
226 							System.err.println("    Fixed: " + line);
227 						}
228 					}
229 
230 					// Keep track of where we are
231 					inDoc = true;
232 					docWasEmpty = true;
233 				} else
234 					System.err.println("Unknown tag: " + line);
235 			}
236 			out.append(line);
237 		} else if (line.indexOf('\t') < 0) {
238 			// no tabs; punctuation
239 			out.append("<pu>").append(StringUtil.escapeXmlChars(line)).append("</pu>");
240 			docWasEmpty = false;
241 		} else {
242 			line = StringUtil.escapeXmlChars(line);
243 			String[] parts = line.split("\t", 3);
244 			out.append("<w p=\"").append(parts[1]).append("\" l=\"").append(parts[2]).append("\">")
245 					.append(parts[0]).append("</w>");
246 			docWasEmpty = false;
247 		}
248 		out.append('\n');
249 		linesDone++;
250 		return shouldContinue;
251 	}

253 	public void convert(Reader in, File outDir, String outFn) throws IOException {
254 		int chunksDone = 0;
255 		Writer out = openOutFile(outDir, outFn, chunksDone);
256 		try {
257 			BufferedReader br;
258 			if (!(in instanceof BufferedReader))
259 				in = new BufferedReader(in);
260 			br = (BufferedReader) in;
261 			while (true) {
262 				String line = br.readLine();
263 				if (line == null)
264 					break;
265 				line = line.trim();
266 				boolean shouldContinueChunk = processLine(line, out);
267 				if (!shouldContinueChunk) {
268 					chunksDone++;
269 					closeOutFile(out);
270 					out = openOutFile(outDir, outFn, chunksDone);
271 				}
272 			}
273 		} finally {
274 			closeOutFile(out);
275 		}
276 	}
277 
278 	private void closeOutFile(Writer out) throws IOException {
279 		if (inDoc) {
280 			out.append("</doc>\n"); // close unclosed doc
281 			System.err.println("Unclosed document at end of chunk");
282 		}
283 		out.append("</docs>\n");
284 		out.close();
285 	}
286 
287 	private Writer openOutFile(File outDir, String outFn, int chunksDone) throws IOException {
288 		linesDone = 0;
289 		if (chunksDone > 0) {
290 			int punt = outFn.lastIndexOf('.');
291 			outFn = outFn.substring(0, punt) + " (" + (chunksDone + 1) + ")"
292 					+ outFn.substring(punt);
293 		}
294 		File outFile = new File(outDir, outFn);
295 		Writer out = new OutputStreamWriter(
296 				new BufferedOutputStream(new FileOutputStream(outFile)), Indexer.DEFAULT_INPUT_ENCODING);
297 		String encName = Indexer.DEFAULT_INPUT_ENCODING.name();
298 		out.append("<?xml version=\"1.0\" encoding=\"" + encName + "\" ?>\n")
299 				.append("<?xml-stylesheet type=\"text/xsl\" href=\"xsl/corpus.xsl\" ?>\n")
300 				.append("<docs file=\"" + outFn + "\">\n");
301 		return out;
302 	}

304 	/**
305 	 * Convert a list of files.
306 	 *
307 	 * @param listFile
308 	 *            list of files to index (assumed to reside in or under basedir)
309 	 * @param inDir
310 	 *            basedir for the files to index
311 	 * @param outDir where to write output files
312 	 * @throws Exception
313 	 */
314 	private static void convertList(File listFile, File inDir, File outDir) throws Exception {
315 		SketchToXmlConverter converter = new SketchToXmlConverter();
316 		List<String> filesToRead = FileUtil.readLines(listFile);
317 		// Contains a list of files to index
318 		for (String filePath : filesToRead) {
319 			File file = new File(inDir, filePath);
320 			convertFile(converter, file, outDir);
321 		}
322 	}
323 
324 	private static void convertFile(SketchToXmlConverter converter, File inFile, File outDir)
325 			throws UnsupportedEncodingException, FileNotFoundException, IOException {
326 		try (Reader in = new InputStreamReader(new FileInputStream(inFile), Indexer.DEFAULT_INPUT_ENCODING)) {
327 			String fn = inFile.getName();
328 			String outFn = fn.substring(0, fn.lastIndexOf('.')) + ".xml";
329 			converter.convert(in, outDir, outFn);
330 		}
331 	}
