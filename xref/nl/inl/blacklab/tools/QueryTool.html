333 	/**
334 	 * The main program.
335 	 *
336 	 * @param args
337 	 *            commandline arguments
338 	 * @throws IOException
339 	 */
340 	public static void main(String[] args) throws IOException {
341 
342 		// Parse command line
343 		File indexDir = null;
344 		File inputFile = null;
345 		String encoding = Charset.defaultCharset().name();
346 		for (int i = 0; i < args.length; i++)  {
347 			String arg = args[i].trim();
348 			if (arg.length() > 0 && arg.charAt(0) == '-') {
349 				if (arg.equals("-e")) {
350 					if (i + 1 == args.length) {
351 						System.err.println("-e option needs argument");
352 						usage();
353 						return;
354 					}
355 					encoding = args[i + 1];
356 					i++;
357 				} else if (arg.equals("-f")) {
358 					if (i + 1 == args.length) {
359 						System.err.println("-f option needs argument");
360 						usage();
361 						return;
362 					}
363 					inputFile = new File(args[i + 1]);
364 					i++;
365 					System.err.println("Batch mode; reading commands from " + inputFile);
366 				} else {
367 					System.err.println("Unknown option: " + arg);
368 					usage();
369 					return;
370 				}
371 			} else {
372 				if (indexDir != null) {
373 					System.err.println("Can only specify 1 index directory");
374 					usage();
375 					return;
376 				}
377 				indexDir = new File(arg);
378 			}
379 		}
380 		if (indexDir == null) {
381 			usage();
382 			return;
383 		}
384 
385 		LogUtil.initLog4jIfNotAlready();
386 		run(indexDir, inputFile, encoding);
387 	}

441 	 * @param indexDir the index to query
442 	 * @param inputFile if specified, run in batch mode. If null, run in interactive mode
443 	 * @param encoding the output encoding to use
444 	 * @throws UnsupportedEncodingException
445 	 * @throws CorruptIndexException
446 	 * @throws IOException
447 	 */
448 	private static void run(File indexDir, File inputFile, String encoding)
449 			throws UnsupportedEncodingException, CorruptIndexException {
450 		if (!indexDir.exists() || !indexDir.isDirectory()) {
451 			System.err.println("Index dir " + indexDir.getPath() + " doesn't exist.");
452 			return;
453 		}
454 
455 		// Use correct output encoding
456 		PrintWriter out, err;
457 		try {
458 			// Yes
459 			out = new PrintWriter(new OutputStreamWriter(System.out, encoding), true);
460 			err = new PrintWriter(new OutputStreamWriter(System.err, encoding), true);
461 			out.println("Using output encoding " + encoding + "\n");
462 		} catch (UnsupportedEncodingException e) {
463 			// Nope; fall back to default
464 			System.err.println("Unknown encoding " + encoding + "; using default");
465 			out = new PrintWriter(new OutputStreamWriter(System.out, Charset.defaultCharset()), true);
466 			err = new PrintWriter(new OutputStreamWriter(System.err, Charset.defaultCharset()), true);
467 		}
468 
469 		BufferedReader in;
470 		if (inputFile == null) {
471 			// No input file specified; use stdin
472 			in = new BufferedReader(new InputStreamReader(System.in, encoding));
473 		} else {
474 			// Open input file
475 			in = FileUtil.openForReading(inputFile, INPUT_FILE_ENCODING);
476 			batchMode = true;
477 		}
478 		try {
479 			QueryTool c = new QueryTool(indexDir, in, out, err);
480 			c.commandProcessor();
481 		} finally {
482 			try {
483 				in.close();
484 			} catch (IOException e) {
485 				throw new RuntimeException(e);
486 			}
487 		}
488 	}

609 	/**
610 	 * Parse and execute commands and queries.
611 	 */
612 	public void commandProcessor() {
613 		printHelp();
614 
615 		while (true) {
616 			Parser parser = parsers.get(currentParserIndex);
617 			String prompt = parser.getPrompt() + "> ";
618 			String cmd;
619 			try {
620 				cmd = readCommand(prompt);
621 			} catch (IOException e1) {
622 				throw new RuntimeException(e1);
623 			}
624 			if (cmd == null || cmd.trim().equals("exit")) {
625 				break;
626 			}
627 
628 			boolean printStat = true;
629 			if (cmd.length() > 0 && cmd.charAt(0) == '-') {
630 				// Command preceded by "-": silent, don't output stats
631 				printStat = false;
632 				cmd = cmd.substring(1).trim();
633 			}
634 
635 			if (cmd.length() > 0 && cmd.charAt(0) == '#') {
636 				// Line starting with "#": comment
637 				if (printStat)
638 					statprintln(cmd);
639 				continue;
640 			}
641 
642 			// Comment after command? Strip.
643 			cmd = cmd.replaceAll("#.+$", "").trim();
644 			if (cmd.length() == 0) {
645 				statprintln(""); // output empty lines in stats
646 				continue; // no actual command on line, skip
647 			}
648

1032 	/**
1033 	 * Parse and execute a query in the current query format.
1034 	 *
1035 	 * @param query
1036 	 *            the query
1037 	 */
1038 	private void parseAndExecuteQuery(String query) {
1039 		Timer t = new Timer();
1040 		try {
1041 
1042 			// See if we want to choose any random words
1043 			if (query.contains("@@")) {
1044 				StringBuffer resultString = new StringBuffer();
1045 				Pattern regex = Pattern.compile("@@[A-Za-z0-9_\\-]+");
1046 				Matcher regexMatcher = regex.matcher(query);
1047 				while (regexMatcher.find()) {
1048 					// You can vary the replacement text for each match on-the-fly
1049 					String wordListName = regexMatcher.group().substring(2);
1050 					List<String> list = wordLists.get(wordListName);
1051 					if (list == null) {
1052 						errprintln("Word list '" + wordListName + "' not found!");
1053 						return;
1054 					}
1055 					int randomIndex = (int)(Math.random() * list.size());
1056 					regexMatcher.appendReplacement(resultString, list.get(randomIndex));
1057 				}
1058 				regexMatcher.appendTail(resultString);
1059 				query = resultString.toString();
1060 			}
1061 
1062 			Parser parser = parsers.get(currentParserIndex);
1063 			TextPattern pattern = parser.parse(query);
1064 			if (pattern == null) {
1065 				errprintln("No query to execute.");
1066 				return;
1067 			}
1068 			pattern = pattern.rewrite();
1069 			if (verbose)
1070 				outprintln("TextPattern: " + pattern.toString());
1071 
1072 			// If the query included filter clauses, use those. Otherwise use the global filter, if any.
1073 			Query filterForThisQuery = parser.getIncludedFilterQuery();
1074 			if (filterForThisQuery == null)
1075 				filterForThisQuery = filterQuery;
1076 			Filter filter = filterForThisQuery == null ? null : new QueryWrapperFilter(filterForThisQuery);
1077 
1078 			// Execute search
1079 			SpanQuery spanQuery = searcher.createSpanQuery(pattern, contentsField, filter);
1080 			if (verbose)
1081 				outprintln("SpanQuery: " + spanQuery.toString(contentsField));
1082 			hits = searcher.find(spanQuery);
1083 			docs = null;
1084 			groups = null;
1085 			collocations = null;
1086 			showWhichGroup = -1;
1087 			showSetting = ShowSetting.HITS;
1088 			firstResult = 0;
1089 			showResultsPage();
1090 			reportTime(t.elapsed());
1091 			if (determineTotalNumberOfHits)
1092 				statInfo = "" + hits.size();
1093 			else
1094 				statInfo = "?";
1095 			commandWasQuery = true;
1096 		} catch (ParseException e) {
1097 			// Parse error
1098 			errprintln("Invalid query: " + e.getMessage());
1099 			errprintln("(Type 'help' for examples or see accompanying documents)");
1100 		} catch (UnsupportedOperationException e) {
1101 			// Unimplemented part of query language used
1102 			errprintln("Cannot execute query; " + e.getMessage());
1103 			errprintln("(Type 'help' for examples or see accompanying documents)");
1104 		}
1105 	}

1274 
1275 		if (!groupBy.equals("hit") && !groupBy.equals("word") && !groupBy.equals("match") && !groupBy.equals("left") && !groupBy.equals("right")) {
1276 			// Assume we want to group by matched text if we don't specify it explicitly.
1277 			property = groupBy;
1278 			groupBy = "match";
1279 		}
1280 
1281 		// Group results
1282 		HitProperty crit = null;
1283 		try {
1284 			if (groupBy.equals("word") || groupBy.equals("match") || groupBy.equals("hit"))
1285 				crit = new HitPropertyHitText(hits, contentsField, property);
1286 			else if (groupBy.startsWith("left"))
1287 				crit = new HitPropertyWordLeft(hits, contentsField, property);
1288 			else if (groupBy.startsWith("right"))
1289 				crit = new HitPropertyWordRight(hits, contentsField, property);
1290 			else if (groupBy.equals("test")) {
1291 				HitProperty p1 = new HitPropertyHitText(hits, contentsField, "lemma");
1292 				HitProperty p2 = new HitPropertyHitText(hits, contentsField, "type");
1293 				crit = new HitPropertyMultiple(p1, p2);
1294 			}
1295 		} catch (Exception e) {
1296 			errprintln("Unknown property: " + property);
1297 			return;
1298 		}
1299 		if (crit == null) {
1300 			errprintln("Unknown criterium: " + groupBy);
1301 			return;
1302 		}
1303 		groups = hits.groupedBy(crit);
1304 		showSetting = ShowSetting.GROUPS;
1305 		sortGroups("size");
1306 		if (property == null)
1307 			property = "(default)";
1308 		reportTime(t.elapsed());
1309 	}
