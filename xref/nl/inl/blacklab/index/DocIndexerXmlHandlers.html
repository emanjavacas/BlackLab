90  		/** Open tag: start indexing this document */
91  		@Override
92  		public void startElement(String uri, String localName, String qName,
93  				Attributes attributes) {
94  			startCaptureContent(contentsField.getName());
95  
96  			currentLuceneDoc = new Document();
97  			currentDocumentName = fileName;
98  			if (currentDocumentName == null)
99  				currentDocumentName = "?";
100 			// Store attribute values from the tag as metadata fields
101 			for (int i = 0; i < attributes.getLength(); i++) {
102 				addMetadataField(attributes.getLocalName(i),
103 						attributes.getValue(i));
104 			}
105 			currentLuceneDoc.add(new Field("fromInputFile", currentDocumentName, indexer.metadataFieldTypeUntokenized));
106 			addMetadataFieldsFromParameters();
107 			indexer.getListener().documentStarted(currentDocumentName);
108 		}
109 
110 		/** Open tag: end indexing the document */
111 		@Override
112 		public void endElement(String uri, String localName, String qName) {
113 			// Make sure all the properties have an equal number of values.
114 			// See what property has the highest position
115 			// (in practice, only starttags and endtags should be able to have
116 			// a position one higher than the rest)
117 			int lastValuePos = 0;
118 			for (ComplexFieldProperty prop : contentsField.getProperties()) {
119 				if (prop.lastValuePosition() > lastValuePos)
120 					lastValuePos = prop.lastValuePosition();
121 			}
122 
123 			// Make sure we always have one more token than the number of
124 			// words, so there's room for any tags after the last word, and we
125 			// know we should always skip the last token when matching.
126 			if (propMain.lastValuePosition() == lastValuePos)
127 				lastValuePos++;
128 
129 			// Add empty values to all lagging properties
130 			for (ComplexFieldProperty prop : contentsField.getProperties()) {
131 				while (prop.lastValuePosition() < lastValuePos) {
132 					prop.addValue("");
133 					if (prop.hasPayload())
134 						prop.addPayload(null);
135 					if (prop == propMain) {
136 						contentsField.addStartChar(getCharacterPosition());
137 						contentsField.addEndChar(getCharacterPosition());
138 					}
139 				}
140 			}
141 
142 			// Finish storing the document in the document store (parts of it
143 			// may already have been written because we write in chunks to save memory),
144 			// retrieve the content id, and store that in Lucene.
145 			// (Note that we do this after adding the dummy token, so the character
146 			//  positions for the dummy token still make (some) sense)
147 			int contentId = storeCapturedContent();
148 			currentLuceneDoc.add(new IntField(ComplexFieldUtil
149 					.contentIdField(contentsField.getName()), contentId,
150 					Store.YES));
151 
152 			// Store the different properties of the complex contents field that
153 			// were gathered in
154 			// lists while parsing.
155 			contentsField.addToLuceneDoc(currentLuceneDoc);
156 
157 
158 			// Add all properties to forward index
159 			for (ComplexFieldProperty prop : contentsField.getProperties()) {
160 				if (!prop.hasForwardIndex())
161 					continue;
162 
163 				// Add property (case-sensitive tokens) to forward index and add
164 				// id to Lucene doc
165 				String propName = prop.getName();
166 				String fieldName = ComplexFieldUtil.propertyField(
167 						contentsField.getName(), propName);
168 				int fiid = indexer.addToForwardIndex(fieldName, prop);
169 				currentLuceneDoc.add(new IntField(ComplexFieldUtil
170 						.forwardIndexIdField(fieldName), fiid, Store.YES));
171 			}
172 
173 			// If there's an external metadata fetcher, call it now so it can
174 			// add the metadata for this document and (optionally) store the
175 			// metadata
176 			// document in the content store (and the corresponding id in the
177 			// Lucene doc)
178 			MetadataFetcher m = getMetadataFetcher();
179 			if (m != null) {
180 				m.addMetadata();
181 			}
182 
183 			// See what metadatafields are missing or empty and add unknown value
184 			// if desired.
185 			IndexStructure struct = indexer.getSearcher().getIndexStructure();
186 			for (String fieldName: struct.getMetadataFields()) {
187 				MetadataFieldDesc fd = struct.getMetadataFieldDesc(fieldName);
188 				boolean missing = false, empty = false;
189 				String currentValue = currentLuceneDoc.get(fieldName);
190 				if (currentValue == null)
191 					missing = true;
192 				else if (currentValue.length() == 0)
193 					empty = true;
194 				UnknownCondition cond = fd.getUnknownCondition();
195 				boolean useUnknownValue = false;
196 				switch(cond) {
197 				case EMPTY:
198 					useUnknownValue = empty;
199 					break;
200 				case MISSING:
201 					useUnknownValue = missing;
202 					break;
203 				case MISSING_OR_EMPTY:
204 					useUnknownValue = missing | empty;
205 					break;
206 				case NEVER:
207 					useUnknownValue = false;
208 					break;
209 				}
210 				if (useUnknownValue)
211 					addMetadataField(fieldName, fd.getUnknownValue());
212 			}
213 
214 			try {
215 				// Add Lucene doc to indexer
216 				indexer.add(currentLuceneDoc);
217 			} catch (Exception e) {
218 				throw ExUtil.wrapRuntimeException(e);
219 			}
220 
221 			// Report progress
222 			reportCharsProcessed();
223 			reportTokensProcessed(wordsDone);
224 			wordsDone = 0;
225 			indexer.getListener().documentDone(currentDocumentName);
226 
227 			// Reset contents field for next document
228 			contentsField.clear();
229 			currentLuceneDoc = null;
230 
231 			// Stop if required
232 			if (!indexer.continueIndexing())
233 				throw new MaxDocsReachedException();
234 		}
235 	}

331 	public class InlineTagHandler extends ElementHandler {
332 
333 		/** Open tag: store the start tag location and the attribute values */
334 		@Override
335 		public void startElement(String uri, String localName, String qName,
336 				Attributes attributes) {
337 			int lastStartTagPos = propStartTag.lastValuePosition();
338 			int currentPos = propMain.lastValuePosition() + 1;
339 			int posIncrement = currentPos - lastStartTagPos;
340 			propStartTag.addValue(localName, posIncrement);
341 			propStartTag.addPayload(null);
342 			int startTagIndex = propStartTag.getLastValueIndex();
343 			openTags.add(new OpenTagInfo(currentPos, startTagIndex));
344 			for (int i = 0; i < attributes.getLength(); i++) {
345 				// Index element attribute values
346 				String name = attributes.getLocalName(i);
347 				String value = attributes.getValue(i);
348 				propStartTag.addValue(
349 						"@" + name.toLowerCase() + "__" + value.toLowerCase(),
350 						0);
351 				propStartTag.addPayload(null);
352 			}
353 		}
354 
355 		/** Close tag: store the end tag location */
356 		@Override
357 		public void endElement(String uri, String localName, String qName) {
358 			int currentPos = propMain.lastValuePosition() + 1;
359 
360 			// Add payload to start tag property indicating end position
361 			OpenTagInfo openTag = openTags.remove(openTags.size() - 1);
362 			byte[] payload = ByteBuffer.allocate(4).putInt(currentPos).array();
363 			propStartTag.setPayloadAtIndex(openTag.index, new BytesRef(payload));
364 		}
365 	}

479 	SensitivitySetting getSensitivitySetting(String propName) {
480 		// See if it's specified in a parameter
481 		String strSensitivity = getParameter(propName + "_sensitivity");
482 		if (strSensitivity != null) {
483 			if (strSensitivity.equals("i"))
484 				return SensitivitySetting.ONLY_INSENSITIVE;
485 			if (strSensitivity.equals("s"))
486 				return SensitivitySetting.ONLY_SENSITIVE;
487 			if (strSensitivity.equals("si") || strSensitivity.equals("is"))
488 				return SensitivitySetting.SENSITIVE_AND_INSENSITIVE;
489 			if (strSensitivity.equals("all"))
490 				return SensitivitySetting.CASE_AND_DIACRITICS_SEPARATE;
491 		}
492 
493 		// Not in parameter (or unrecognized value), use default based on
494 		// propName
495 		if (propName.equals(ComplexFieldUtil.getDefaultMainPropName())
496 				|| propName.equals(ComplexFieldUtil.LEMMA_PROP_NAME)) {
497 			// Word: default to sensitive/insensitive
498 			return SensitivitySetting.SENSITIVE_AND_INSENSITIVE;
499 		}
500 		if (propName.equals(ComplexFieldUtil.PUNCTUATION_PROP_NAME)) {
501 			// Punctuation: default to only insensitive
502 			return SensitivitySetting.ONLY_INSENSITIVE;
503 		}
504 		if (propName.equals(ComplexFieldUtil.START_TAG_PROP_NAME)
505 				|| propName.equals(ComplexFieldUtil.END_TAG_PROP_NAME)) {
506 			// XML tag properties: default to only sensitive
507 			return SensitivitySetting.ONLY_SENSITIVE;
508 		}
509 
510 		// Unrecognized; default to only insensitive
511 		return SensitivitySetting.ONLY_INSENSITIVE;
512 	}

547 	/**
548 	 * StringBuffer re-used for building start/end tags and processing
549 	 * instructions.
550 	 */
551 	StringBuilder elementBuilder = new StringBuilder();
552 
553 	public void startElement(String uri, String localName, String qName,
554 			Attributes attributes) {
555 		// Call any hooks associated with this element
556 		hookableHandler.startElement(uri, localName, qName, attributes);
557 
558 		elementBuilder.setLength(0); // clear
559 		elementBuilder.append("<").append(qName);
560 		for (int i = 0; i < attributes.getLength(); i++) {
561 			String value = escapeXmlChars(attributes.getValue(i));
562 			elementBuilder.append(" ").append(attributes.getQName(i))
563 					.append("=\"").append(value).append("\"");
564 		}
565 		// Append any namespace mapping not yet outputted
566 		if (!outputPrefixMapping.isEmpty()) {
567 			for (Map.Entry<String, String> e : outputPrefixMapping.entrySet()) {
568 				if (e.getKey().length() == 0)
569 					elementBuilder.append(" xmlns=\"").append(e.getValue())
570 							.append("\"");
571 				else
572 					elementBuilder.append(" xmlns:").append(e.getKey())
573 							.append("=\"").append(e.getValue()).append("\"");
574 			}
575 			outputPrefixMapping.clear(); // outputted all prefix mappings for
576 											// now
577 		}
578 		elementBuilder.append(">");
579 		processContent(elementBuilder.toString());
580 	}

583 	 * StringBuffer re-used for escaping XML chars
584 	 */
585 	StringBuilder escapeBuilder = new StringBuilder();
586 
587 	/**
588 	 * Escape the special XML chars (<, >, &, ") with their named entity
589 	 * equivalents.
590 	 *
591 	 * NOTE: copy of StringUtil.escapeXmlChars that re-uses its StringBuilder
592 	 * for increased memory efficiency.
593 	 *
594 	 * @param source
595 	 *            the source string
596 	 * @return the escaped string
597 	 */
598 	public String escapeXmlChars(String source) {
599 		escapeBuilder.setLength(0); // clear
600 		int start = 0;
601 		char[] srcArr = new char[source.length()];
602 		source.getChars(0, source.length(), srcArr, 0);
603 		int end = source.length();
604 		for (int i = 0; i < end; i++) {
605 			char c = srcArr[i]; // source.charAt(i);
606 			if (c == '<' || c == '>' || c == '&' || c == '"') {
607 				escapeBuilder.append(srcArr, start, i - start); // source.substring(start,
608 																// i));
609 				switch (c) {
610 				case '<':
611 					escapeBuilder.append("&lt;");
612 					break;
613 				case '>':
614 					escapeBuilder.append("&gt;");
615 					break;
616 				case '&':
617 					escapeBuilder.append("&amp;");
618 					break;
619 				case '"':
620 					escapeBuilder.append("&quot;");
621 					break;
622 				}
623 				start = i + 1;
624 			}
625 		}
626 		escapeBuilder.append(srcArr, start, end - start); // source.substring(start));
627 		return escapeBuilder.toString();
628 	}

631 	 * Escape the special XML chars (<, >, &, ") with their named entity
632 	 * equivalents.
633 	 *
634 	 * NOTE: copy of StringUtil.escapeXmlChars that re-uses its StringBuilder
635 	 * for increased memory efficiency.
636 	 *
637 	 * @param source
638 	 *            the source string
639 	 * @param start
640 	 *            start index of the string to escape
641 	 * @param length
642 	 *            length of the string to escape
643 	 * @return the escaped string
644 	 */
645 	public String escapeXmlChars(char[] source, int start, int length) {
646 		escapeBuilder.setLength(0); // clear
647 		int end = start + length;
648 		for (int i = start; i < end; i++) {
649 			char c = source[i];
650 			if (c == '<' || c == '>' || c == '&' || c == '"') {
651 				escapeBuilder.append(source, start, i - start);
652 				switch (c) {
653 				case '<':
654 					escapeBuilder.append("&lt;");
655 					break;
656 				case '>':
657 					escapeBuilder.append("&gt;");
658 					break;
659 				case '&':
660 					escapeBuilder.append("&amp;");
661 					break;
662 				case '"':
663 					escapeBuilder.append("&quot;");
664 					break;
665 				}
666 				start = i + 1;
667 			}
668 		}
669 		escapeBuilder.append(source, start, end - start);
670 		return escapeBuilder.toString();
671 	}

827 	public void addMetadataField(String name, String value) {
828 
829 		IndexStructure struct = indexer.getSearcher().getIndexStructure();
830 		struct.registerMetadataField(name);
831 
832 		MetadataFieldDesc desc = struct.getMetadataFieldDesc(name);
833 		FieldType type = desc.getType();
834 		desc.addValue(value);
835 
836 		FieldType shouldBeType = getMetadataFieldTypeFromIndexerProperties(name);
837 		if (type == FieldType.TEXT
838 				&& shouldBeType != FieldType.TEXT) {
839 			// indexer.properties overriding default type
840 			type = shouldBeType;
841 		}
842 		if (type != FieldType.NUMERIC) {
843 			currentLuceneDoc.add(new Field(name, value,
844 					luceneTypeFromIndexStructType(type)));
845 		}
846 		if (type == FieldType.NUMERIC || numericFields.contains(name)) {
847 			String numFieldName = name;
848 			if (type != FieldType.NUMERIC) {
849 				numFieldName += "Numeric";
850 			}
851 			// Index these fields as numeric too, for faster range queries
852 			// (we do both because fields sometimes aren't exclusively numeric)
853 			int n = 0;
854 			try {
855 				n = Integer.parseInt(value);
856 			} catch (NumberFormatException e) {
857 				// This just happens sometimes, e.g. given multiple years, or
858 				// descriptive text like "around 1900". OK to ignore.
859 			}
860 			IntField nf = new IntField(numFieldName, n, Store.YES);
861 			currentLuceneDoc.add(nf);
862 		}
863 	}

898 	@Override
899 	public void index() throws IOException, InputFormatException {
900 		SAXParserFactory factory = SAXParserFactory.newInstance();
901 		factory.setNamespaceAware(true);
902 		SAXParser parser;
903 		try {
904 			parser = factory.newSAXParser();
905 		} catch (Exception e1) {
906 			// Unrecoverable error, throw runtime exception
907 			throw new RuntimeException(e1);
908 		}
909 		try {
910 			InputSource is = new InputSource(reader);
911 			XMLReader xmlReader = parser.getXMLReader();
912 			xmlReader.setEntityResolver(new DummyEntityResolver());
913 			xmlReader.setContentHandler(saxParseHandler);
914 			xmlReader.parse(is);
915 		} catch (IOException e) {
916 			throw e;
917 		} catch (SAXException e) {
918 			throw new InputFormatException(e);
919 		} catch (DocIndexer.MaxDocsReachedException e) {
920 			// OK; just stop indexing prematurely
921 		}
922 
923 		if (nDocumentsSkipped > 0)
924 			System.err.println("Skipped " + nDocumentsSkipped
925 					+ " large documents");
926 	}
