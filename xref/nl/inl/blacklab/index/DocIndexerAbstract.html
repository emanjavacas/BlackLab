27  /**
28   * Abstract base class for a DocIndexer processing XML files.
29   */
30  public abstract class DocIndexerAbstract implements DocIndexer {
31  	/**
32  	 * Write content chunks per 10M (i.e. don't keep all content in memory at all times)
33  	 */
34  	private static final long WRITE_CONTENT_CHUNK_SIZE = 10000000;
35  
36  	protected boolean skippingCurrentDocument = false;
37  
38  	public Indexer indexer;
39  
40  	/**
41  	 * File we're currently parsing. This can be useful for storing the original filename in the
42  	 * index.
43  	 */
44  	public String fileName;
45  
46  	protected CountingReader reader;
47  
48  	//protected ContentStore contentStore;
49  
50  	private StringBuilder content = new StringBuilder();
51  
52  	/** Are we capturing the content of the document for indexing? */
53  	private boolean captureContent = false;
54  
55  	/** What field we're capturing content for */
56  	private String captureContentFieldName;
57  
58  	private int charsContentAlreadyStored = 0;
59  
60  	protected int nDocumentsSkipped = 0;
61  
62  	/** Do we want to omit norms? (Default: yes) */
63  	public boolean omitNorms = true;
64  
65  	/**
66  	 * Returns our Indexer object
67  	 * @return the Indexer object
68  	 */
69  	public Indexer getIndexer() {
70  		return indexer;
71  	}
72  
73  	/**
74  	 * Enables or disables norms. Norms are disabled by default.
75  	 *
76  	 * The method name was chosen to match Lucene's Field.setOmitNorms().
77  	 * Norms are only required if you want to use document-length-normalized scoring.
78  	 *
79  	 * @param b if true, doesn't store norms; if false, does store norms
80  	 */
81  	public void setOmitNorms(boolean b) {
82  		omitNorms = b;
83  	}
84  
85  	boolean continueIndexing() {
86  		return indexer.continueIndexing();
87  	}
88  
89  	/**
90  	 * Starts capturing of content.
91  	 * @deprecated pass the field name you're capturing content for
92  	 */
93  	@Deprecated
94  	public void startCaptureContent() {
95  		startCaptureContent(Searcher.DEFAULT_CONTENTS_FIELD_NAME);
96  	}
97  
98  	public void startCaptureContent(String fieldName) {
99  		captureContent = true;
100 		captureContentFieldName = fieldName;
101 
102 		// Empty the StringBuilder object
103 		content.setLength(0);
104 	}
105 
106 	public int storeCapturedContent() {
107 		captureContent = false;
108 		int id = -1;
109 		if (!skippingCurrentDocument) {
110 			ContentStore contentStore = indexer.getContentStore(captureContentFieldName);
111 			id = contentStore.store(content.toString());
112 		}
113 		content.setLength(0);
114 		charsContentAlreadyStored = 0;
115 		return id;
116 	}
117 
118 	public void storePartCapturedContent() {
119 		charsContentAlreadyStored += content.length();
120 		if (!skippingCurrentDocument) {
121 			ContentStore contentStore = indexer.getContentStore(captureContentFieldName);
122 			contentStore.storePart(content.toString());
123 		}
124 		content.setLength(0);
125 	}
126 
127 	private void appendContentInternal(String str) {
128 		content.append(str);
129 	}
130 
131 	public void appendContent(String str) {
132 		appendContentInternal(str);
133 		if (content.length() >= WRITE_CONTENT_CHUNK_SIZE) {
134 			storePartCapturedContent();
135 		}
136 	}
137 
138 	public void appendContent(char[] buffer, int start, int length) {
139 		appendContentInternal(new String(buffer, start, length));
140 		if (content.length() >= WRITE_CONTENT_CHUNK_SIZE) {
141 			storePartCapturedContent();
142 		}
143 	}
144 
145 	public void processContent(char[] buffer, int start, int length) {
146 		if (captureContent)
147 			appendContent(buffer, start, length);
148 	}
149 
150 	public void processContent(String contentToProcess) {
151 		if (captureContent)
152 			appendContent(contentToProcess);
153 	}
154 
155 	/**
156 	 * Returns the current position in the original XML content in chars.
157 	 * @return the current char position.
158 	 * @deprecated renamed to getCharacterPositionInContent() for clarity
159 	 */
160 	@Deprecated
161 	public int getContentPosition() {
162 		return charsContentAlreadyStored + content.length();
163 	}
164 
165 	/**
166 	 * Returns the current position in the original XML content in chars.
167 	 * @return the current char position
168 	 */
169 	public int getCharacterPosition() {
170 		return charsContentAlreadyStored + content.length();
171 	}

173 	/**
174 	 * Provided for compatibility with Meertens' fork of BlackLab;
175 	 * will eventually be removed in favor of an "official" way
176 	 * to index content without storing in a ContentStore.
177 	 *
178 	 * Not sure if this will always give the correct position because
179 	 * the original input is used for highlighting, not the reconstructed
180 	 * XML in the content variable.
181 	 *
182 	 * @return the content length
183 	 * @deprecated will be handled differently in the future.
184 	 */
185 	@Deprecated
186 	public int getContentPositionNoStore(){
187 		return content.length();
188 	}

190 	public DocIndexerAbstract(Indexer indexer, String fileName, Reader reader) {
191 		this.indexer = indexer;
192 		this.fileName = fileName;
193 		this.reader = new CountingReader(reader);
194 
195 		// Get our parameters from the indexer
196 		Map<String, String> indexerParameters = indexer.getIndexerParameters();
197 		if (indexerParameters != null)
198 			setParameters(indexerParameters);
199 	}
200 
201 	public void reportCharsProcessed() {
202 		long charsProcessed = reader.getCharsReadSinceLastCall();
203 		indexer.getListener().charsDone(charsProcessed);
204 	}
205 
206 	public void reportTokensProcessed(int n) {
207 		indexer.getListener().tokensDone(n);
208 	}
209 
210 	/**
211 	 * Parameters passed to this indexer
212 	 */
213 	protected Map<String, String> parameters = new HashMap<>();
214 
215 	@Override
216 	public boolean hasParameter(String name) {
217 		return parameters.containsKey(name);
218 	}
219 
220 	@Override
221 	public void setParameter(String name, String value) {
222 		parameters.put(name, value);
223 	}
224 
225 	@Override
226 	public void setParameters(Map<String, String> param) {
227 		for (Map.Entry<String, String> e: param.entrySet()) {
228 			parameters.put(e.getKey(), e.getValue());
229 		}
230 	}
231 
232 	@Override
233 	public String getParameter(String name, String defaultValue) {
234 		String value = parameters.get(name);
235 		if (value == null)
236 			return defaultValue;
237 		return value;
238 	}
239 
240 	@Override
241 	public String getParameter(String name) {
242 		return getParameter(name, null);
243 	}
244 
245 	public boolean getParameter(String name, boolean defaultValue) {
246 		String value = parameters.get(name);
247 		if (value == null)
248 			return defaultValue;
249 		value = value.trim().toLowerCase();
250 		return value.equals("true") || value.equals("1") || value.equals("yes");
251 	}
252 
253 	public int getParameter(String name, int defaultValue) {
254 		String value = parameters.get(name);
255 		if (value == null)
256 			return defaultValue;
257 		try {
258 			return Integer.parseInt(value);
259 		} catch (NumberFormatException e) {
260 			return defaultValue;
261 		}
262 	}
263 
264 	protected boolean tokenizeField(String name) {
265 		String parName = name + "_tokenized";
266 		if (!hasParameter(name + "_tokenized")) {
267 			parName = name + "_analyzed"; // Check the old (Lucene 3.x) term, "analyzed"
268 		}
269 
270 		return getParameter(parName, true);
271 	}

280 	@Override
281 	public FieldType getMetadataFieldTypeFromIndexerProperties(String fieldName) {
282 		if (tokenizeField(fieldName))
283 			return FieldType.TEXT;
284 		return FieldType.UNTOKENIZED;
285 	}
286 
287 	protected org.apache.lucene.document.FieldType luceneTypeFromIndexStructType(FieldType type) {
288 		switch (type) {
289 		case NUMERIC:
290 			throw new IllegalArgumentException("Numeric types should be indexed using IntField, etc.");
291 		case TEXT:
292 			return indexer.metadataFieldTypeTokenized;
293 		case UNTOKENIZED:
294 			return indexer.metadataFieldTypeUntokenized;
295 		default:
296 			throw new IllegalArgumentException("Unknown field type: " + type);
297 		}
298 	}
