18  /**
19   * Document format registry, for resolving a DocIndexer class given a
20   * format identifier (common abbreviation or (qualified) class name).
21   */
22  public class DocumentFormats {
23  
24  	/** Document formats */
25  	static Map<String, Class<? extends DocIndexer>> formats = new TreeMap<>();
26  
27  	static {
28  		// Some abbreviations for commonly used builtin DocIndexers.
29  		// You can also specify the classname for builtin DocIndexers,
30  		// or a fully-qualified name for your custom DocIndexer (must
31  		// be on the classpath)
32  		register("alto", DocIndexerAlto.class);
33  		register("folia", DocIndexerFolia.class);
34  		register("whitelab2", DocIndexerWhiteLab2.class);
35  		register("pagexml", DocIndexerPageXml.class);
36  		register("sketchxml", DocIndexerXmlSketch.class);
37  
38  		// TEI has a number of variants
39  		// By default, the contents of the "body" element are indexed, but alternatively you can index the contents of "text".
40  		// By default, the "type" attribute is assumed to contain PoS, but alternatively you can use the "function" attribute.
41  		register("tei", DocIndexerTei.class);
42  		register("tei-element-body", DocIndexerTei.class);
43  		register("tei-element-text", DocIndexerTeiText.class);
44  		register("tei-pos-type", DocIndexerTei.class);
45  		register("tei-pos-function", DocIndexerTeiPosInFunctionAttr.class);
46  	}
47  
48  	/**
49  	 * Register an abbreviated document format identifier.
50  	 *
51  	 * @param formatAbbreviation the format abbreviation, e.g. "tei"
52  	 *   (NOTE: format abbreviations are case-insensitive, and are lowercased internally)
53  	 * @param docIndexerClass the DocIndexer class for this format
54  	 */
55  	public static void register(String formatAbbreviation, Class<? extends DocIndexer> docIndexerClass) {
56  		formats.put(formatAbbreviation.toLowerCase(), docIndexerClass);
57  	}
58  
59  	/**
60  	 * Get the DocIndexer class associated with the format identifier.
61  	 *
62  	 * @param formatIdentifier format identifier, e.g. "tei" or "com.example.MyIndexer"
63  	 * @return the DocIndexer class, or null if not found
64  	 */
65  	public static Class<? extends DocIndexer> getIndexerClass(String formatIdentifier) {
66  		// Check if it's a known abbreviation.
67  		Class<?> docIndexerClass = formats.get(formatIdentifier.toLowerCase());
68  		if (docIndexerClass == null) {
69  			// No; is it a fully qualified class name?
70  			try {
71  				docIndexerClass = Class.forName(formatIdentifier);
72  			} catch (Exception e1) {
73  				try {
74  					// No. Is it a class in the BlackLab indexers package?
75  					docIndexerClass = Class.forName("nl.inl.blacklab.indexers." + formatIdentifier);
76  				} catch (Exception e) {
77  					// Couldn't be resolved. That's okay, we'll just return null and let
78  					// the application deal with it.
79  				}
80  			}
81  		}
82  		return docIndexerClass.asSubclass(DocIndexer.class);
83  	}
84  
85  	/**
86  	 * Check if a particular string denotes a valid document format.
87  	 *
88  	 * @param formatIdentifier format identifier, e.g. "tei" or "com.example.MyIndexer"
89  	 * @return true iff it corresponds to a format
90  	 */
91  	public static boolean exists(String formatIdentifier) {
92  		return getIndexerClass(formatIdentifier) != null;
93  	}
94  
95  	/**
96  	 * Returns a sorted list of registered document format abbreviations.
97  	 *
98  	 * @return the list of registered abbreviations
99  	 */
100 	public static List<String> list() {
101 		List<String> l = new ArrayList<>(formats.keySet());
102 		Collections.sort(l);
103 		return Collections.unmodifiableList(l);
104 	}
105 
106 }
