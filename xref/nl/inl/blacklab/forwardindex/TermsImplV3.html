33  /**
34   * Keeps a first-come-first-serve list of unique terms.
35   * Each term gets a unique index number. These numbers are
36   * stored in the forward index to conserve space and allow quick
37   * lookups of terms occurring in specific positions.
38   *
39   * This version of the class stores the terms in a more efficient way so it
40   * saves and loads faster, and includes the case-insensitive sorting order.
41   */
42  class TermsImplV3 extends Terms {
43  	private static final int APPROX_MAX_ARRAY_SIZE = Integer.MAX_VALUE - 100;
44  
45  	/** Number of sort buffers we store in the terms file (case-sensitive/insensitive and inverted buffers for both as well) */
46  	private static final int NUM_SORT_BUFFERS = 4;
47  
48  	/** Number of bytes per int */
49  	private static final int BYTES_PER_INT = Integer.SIZE / Byte.SIZE;
50  
51  	protected static final Logger logger = Logger.getLogger(TermsImplV3.class);
52  
53  	/** Search mode only: the terms, by index number. */
54  	String[] terms;
55  
56  	/** The sorting position for each index number. Inverse of idPerSortPosition[]
57  	 *  array. Only valid when indexMode == false. */
58  	int[] sortPositionPerId;
59  
60  	/** The case-insensitive sorting position for each index number. Inverse of idPerSortPosition[]
61  	 *  array. Only valid when indexMode == false. */
62  	int[] sortPositionPerIdInsensitive;
63  
64  	/**
65  	 * Mapping from term to its unique index number. We use a SortedMap because we wish to
66  	 * store the sorted index numbers later (to speed up sorting). Only valid in indexMode.
67  	 */
68  	Map<String, Integer> termIndex;
69  
70  	/** If true, we're indexing data and adding terms. If false, we're searching and just retrieving terms. */
71  	private boolean indexMode;
72  
73  	/** If true, termIndex is a valid mapping from term to term id. */
74  	private boolean termIndexBuilt;
75  
76  	/**
77  	 * Collator to use for string comparisons
78  	 */
79  	final Collator collator;
80  
81  	/**
82  	 * Collator to use for insensitive string comparisons
83  	 */
84  	Collator collatorInsensitive;
85  
86  	/** Use new blocks-based terms file, that can grow larger than 2 GB? */
87  	private boolean useBlockBasedTermsFile = true;
88  
89  	TermsImplV3(boolean indexMode, Collator collator, File termsFile, boolean useBlockBasedTermsFile) {
90  		this.indexMode = indexMode;
91  		if (collator == null)
92  			collator = Collator.getInstance();
93  		this.collator = (Collator)collator.clone();
94  		this.collator.setStrength(Collator.TERTIARY);
95  		this.collatorInsensitive = (Collator)collator.clone();
96  		collatorInsensitive.setStrength(Collator.PRIMARY);
97  
98  		if (indexMode) {
99  			// Index mode: create a SortedMap based on the specified Collator.
100 			// (used later to get the terms in sort order)
101 			this.termIndex = new TreeMap<>(this.collator);
102 		} else {
103 			// We already have the sort order, so TreeMap is not necessary here.
104 			this.termIndex = Maps.mutable.empty();
105 		}
106 		termIndexBuilt = true;
107 		setBlockBasedFile(useBlockBasedTermsFile);
108 		if (termsFile != null && termsFile.exists())
109 			read(termsFile);
110 	}

112 	@Override
113 	public int indexOf(String term) {
114 
115 		if (!termIndexBuilt) {
116 			// We havent' filled termIndex based on terms[] yet.
117 			// Do so now. (so the first call to this method might be
118 			// slow in search mode, but it's only used to deserialize
119 			// HitPropValueContext*, which doesn't happen a lot)
120 			buildTermIndex();
121 		}
122 
123 		Integer index = termIndex.get(term);
124 		if (index != null)
125 			return index;
126 		if (!indexMode)
127 			return NO_TERM; // term not found
128 		index = termIndex.size();
129 		termIndex.put(term, index);
130 		return index;
131 	}
132 
133 	@Override
134 	public synchronized void buildTermIndex() {
135 		if (termIndexBuilt)
136 			return;
137 		for (int i = 0; i < terms.length; i++) {
138 			termIndex.put(terms[i], i);
139 		}
140 		termIndexBuilt = true;
141 	}
142 
143 	@Override
144 	public void clear() {
145 		if (!indexMode)
146 			throw new RuntimeException("Cannot clear, not in index mode");
147 		termIndex.clear();
148 		termIndexBuilt = true;
149 	}

151 	private void read(File termsFile) {
152 		termIndex.clear();
153 		try {
154 			try (RandomAccessFile raf = new RandomAccessFile(termsFile, "r")) {
155 				try (FileChannel fc = raf.getChannel()) {
156 					MappedByteBuffer buf = fc.map(MapMode.READ_ONLY, 0, termsFile.length());
157 					int n = buf.getInt();
158 					IntBuffer ib = buf.asIntBuffer();
159 					int[] termStringOffsets = new int[n + 1];
160 					terms = new String[n];
161 
162 					if (useBlockBasedTermsFile) {
163 						// New format, multiple blocks of term strings if necessary,
164 						// so term strings may total over 2 GB.
165 
166 						// Read the term string offsets and string data block
167 						int currentTerm = 0;
168 						while (currentTerm < n) {
169 							int numTermsThisBlock = ib.get();
170 							ib.get(termStringOffsets, currentTerm, numTermsThisBlock); // term string offsets
171 
172 							// Read term strings data
173 							int dataBlockSize = termStringOffsets[currentTerm + numTermsThisBlock] = ib.get();
174 							buf.position(buf.position() + BYTES_PER_INT * (numTermsThisBlock + 2));
175 							byte[] termStringsThisBlock = new byte[dataBlockSize];
176 							buf.get(termStringsThisBlock);
177 
178 							// Now instantiate String objects from the offsets and byte data
179 							for ( ; currentTerm < n; currentTerm++) {
180 								int offset = termStringOffsets[currentTerm];
181 								int length = termStringOffsets[currentTerm + 1] - offset;
182 								String str = new String(termStringsThisBlock, offset, length, DEFAULT_CHARSET);
183 
184 								// We need to find term for id while searching
185 								terms[currentTerm] = str;
186 							}
187 
188 							ib = buf.asIntBuffer();
189 						}
190 
191 					} else {
192 						// Old format, single term strings block.
193 						// Causes problems when term strings total over 2 GB.
194 
195 						ib.get(termStringOffsets); // term string offsets
196 						int termStringsByteSize = ib.get(); // data block size
197 
198 						// termStringByteSize fits in an int, and terms
199 						// fits in a single byte array. Use the old code.
200 						buf.position(buf.position() + BYTES_PER_INT + BYTES_PER_INT * termStringOffsets.length);
201 						byte[] termStrings = new byte[termStringsByteSize];
202 						buf.get(termStrings);
203 						ib = buf.asIntBuffer();
204 
205 						// Now instantiate String objects from the offsets and byte data
206 						terms = new String[n];
207 						for (int id = 0; id < n; id++) {
208 							int offset = termStringOffsets[id];
209 							int length = termStringOffsets[id + 1] - offset;
210 							String str = new String(termStrings, offset, length, DEFAULT_CHARSET);
211 
212 							// We need to find term for id while searching
213 							terms[id] = str;
214 						}
215 					}
216 
217 					if (indexMode) {
218 						termIndexBuilt = false;
219 						buildTermIndex(); // We need to find id for term while indexing
220 						terms = null; // useless in index mode because we can't add to it, and we don't need it anyway
221 					} else {
222 						termIndexBuilt = false; // termIndex hasn't been filled yet
223 
224 						// Read the sort order arrays
225 						sortPositionPerId = new int[n];
226 						sortPositionPerIdInsensitive = new int[n];
227 						ib.position(ib.position() + n); // Advance past unused sortPos -> id array (left in there for file compatibility)
228 						ib.get(sortPositionPerId);
229 						ib.position(ib.position() + n); // Advance past unused sortPos -> id array (left in there for file compatibility)
230 						ib.get(sortPositionPerIdInsensitive);
231 					}
232 				}
233 			}
234 		} catch (Exception e) {
235 			throw new RuntimeException(e);
236 		}
237 	}

239 	@Override
240 	public void write(File termsFile) {
241 		if (!indexMode)
242 			throw new RuntimeException("Term.write(): not in index mode!");
243 
244 		try {
245 			// Open the terms file
246 			try (RandomAccessFile raf = new RandomAccessFile(termsFile, "rw")) {
247 				try (FileChannel fc = raf.getChannel()) {
248 					int n = termIndex.size();
249 
250 					// Fill the terms[] array
251 					terms = new String[n];
252 					long termStringsByteSize = 0;
253 					for (Map.Entry<String, Integer> entry: termIndex.entrySet()) {
254 						terms[entry.getValue()] = entry.getKey();
255 						termStringsByteSize += entry.getKey().getBytes(DEFAULT_CHARSET).length;
256 					}
257 
258 					// Calculate the file length and map the file
259 					long fileLength = 2 * BYTES_PER_INT + (n + 1) * BYTES_PER_INT + termStringsByteSize + NUM_SORT_BUFFERS * BYTES_PER_INT * n;
260 					fc.truncate(fileLength); // truncate if necessary
261 					MappedByteBuffer buf = fc.map(MapMode.READ_WRITE, 0, fileLength);
262 					buf.putInt(n); // Start with the number of terms
263 
264 					IntBuffer ib = buf.asIntBuffer();
265 					if (!useBlockBasedTermsFile) {
266 						// Terms file is small enough to fit in a single byte array.
267 						// Use the old code.
268 
269 						// Calculate byte offsets for all the terms and fill data array
270 						int currentOffset = 0;
271 						int[] termStringOffsets = new int[n + 1];
272 						byte[] termStrings = new byte[(int)termStringsByteSize];
273 						for (int i = 0; i < n; i++) {
274 							termStringOffsets[i] = currentOffset;
275 							byte[] termBytes = terms[i].getBytes(DEFAULT_CHARSET);
276 							System.arraycopy(termBytes, 0, termStrings, currentOffset, termBytes.length);
277 							currentOffset += termBytes.length;
278 						}
279 						termStringOffsets[n] = currentOffset;
280 
281 						// Write offset and data arrays to file
282 						ib.put(termStringOffsets);
283 						ib.put((int)termStringsByteSize); // size of the data block to follow
284 						buf.position(buf.position() + BYTES_PER_INT + BYTES_PER_INT * termStringOffsets.length); // advance past offsets array
285 						buf.put(termStrings);
286 						ib = buf.asIntBuffer();
287 					} else {
288 						// Terms file is too large to fit in a single byte array.
289 						// Use the new code.
290 						int currentTerm = 0;
291 						long bytesLeftToWrite = termStringsByteSize;
292 						int[] termStringOffsets = new int[n];
293 						while (currentTerm < n) {
294 							int firstTermInBlock = currentTerm;
295 							int blockSize = (int)Math.min(bytesLeftToWrite, APPROX_MAX_ARRAY_SIZE);
296 
297 							// Calculate byte offsets for all the terms and fill data array
298 							int currentOffset = 0;
299 							byte[] termStrings = new byte[blockSize];
300 							while (currentTerm < n) {
301 								termStringOffsets[currentTerm] = currentOffset;
302 								byte[] termBytes = terms[currentTerm].getBytes(DEFAULT_CHARSET);
303 								if (currentOffset + termBytes.length > blockSize) {
304 									// Block is full. Write it and continue with next block.
305 									break;
306 								}
307 								System.arraycopy(termBytes, 0, termStrings, currentOffset, termBytes.length);
308 								currentOffset += termBytes.length;
309 								currentTerm++;
310 								bytesLeftToWrite -= termBytes.length;
311 							}
312 							// Write offset and data arrays to file
313 							int numTermsThisBlock = currentTerm - firstTermInBlock;
314 							ib.put(numTermsThisBlock);
315 							ib.put(termStringOffsets, firstTermInBlock, numTermsThisBlock);
316 							ib.put(currentOffset); // include the offset after the last term at position termStringOffsets[n]
317 							                       // (doubles as the size of the data block to follow)
318 							buf.position(buf.position() + BYTES_PER_INT * (2 + numTermsThisBlock)); // advance past offsets array
319 							buf.put(termStrings, 0, currentOffset);
320 							ib = buf.asIntBuffer();
321 						}
322 
323 					}
324 
325 					// Write the case-sensitive sort order
326 					// Because termIndex is a SortedMap, values are returned in key-sorted order.
327 					// In other words, the index numbers are in order of sorted terms, so the id
328 					// for 'aardvark' comes before the id for 'ape', etc.
329 					int i = 0;
330 					sortPositionPerId = new int[n];
331 					Integer[] insensitive = new Integer[n];
332 					for (int id: termIndex.values()) {
333 						sortPositionPerId[id] = i;
334 						insensitive[i] = id; // fill this so we can re-sort later, faster b/c already partially sorted
335 						i++;
336 					}
337 					ib.put(new int[n]); // NOT USED ANYMORE, JUST FOR FILE COMPATIBILITY
338 					ib.put(sortPositionPerId);
339 
340 					// Now, sort case-insensitively and write those arrays as well
341 					Arrays.sort(insensitive, new Comparator<Integer>() {
342 						@Override
343 						public int compare(Integer a, Integer b) {
344 							return collatorInsensitive.compare(terms[a], terms[b]);
345 						}
346 					});
347 					// Copy into the sortPositionPerIdInsensitive array, making sure that
348 					// identical values get identical sort positions!
349 					sortPositionPerIdInsensitive = new int[n];
350 					int sortPos = 0;
351 					for (i = 0; i < n; i++) {
352 						if (i == 0 || collatorInsensitive.compare(terms[insensitive[i - 1]], terms[insensitive[i]]) != 0) {
353 							// Not identical to previous value: gets its own sort position.
354 							// If a value is identical to the previous one, it gets the same sort position.
355 							sortPos = i;
356 						}
357 						sortPositionPerIdInsensitive[insensitive[i]] = sortPos;
358 					}
359 					ib.put(new int[n]); // NOT USED ANYMORE, JUST FOR FILE COMPATIBILITY
360 					ib.put(sortPositionPerIdInsensitive);
361 				}
362 			}
363 		} catch (Exception e) {
364 			throw new RuntimeException(e);
365 		}
366 	}

368 	@Override
369 	public String get(Integer index) {
370 		assert index >= 0 && index < terms.length : "Term index out of range (" + index + ", numterms = " + terms.length + ")";
371 		return terms[index];
372 	}

374 	@Override
375 	public int numberOfTerms() {
376 		return sortPositionPerId.length;
377 	}
378 
379 	@Override
380 	public void toSortOrder(int[] tokenId, int[] sortOrder, boolean sensitive) {
381 		if (sensitive) {
382 			for (int i = 0; i < tokenId.length; i++) {
383 				if (tokenId[i] == NO_TERM)
384 					sortOrder[i] = NO_TERM;
385 				else
386 					sortOrder[i] = sortPositionPerId[tokenId[i]];
387 			}
388 		} else {
389 			for (int i = 0; i < tokenId.length; i++) {
390 				if (tokenId[i] == NO_TERM)
391 					sortOrder[i] = NO_TERM;
392 				else
393 					sortOrder[i] = sortPositionPerIdInsensitive[tokenId[i]];
394 			}
395 		}
396 	}
397 
398 	@Override
399 	public int compareSortPosition(int tokenId1, int tokenId2, boolean sensitive) {
400 		if (sensitive) {
401 			return sortPositionPerId[tokenId1] - sortPositionPerId[tokenId2];
402 		}
403 		return sortPositionPerIdInsensitive[tokenId1] - sortPositionPerIdInsensitive[tokenId2];
404 	}
405 
406 	@Override
407 	public int idToSortPosition(int id, boolean sensitive) {
408 		return sensitive ? sortPositionPerId[id] : sortPositionPerIdInsensitive[id];
409 	}

413 		this.useBlockBasedTermsFile = useBlockBasedTermsFile;
