80  	/**
81  	 * Retrieve one or more substrings from the specified content.
82  	 *
83  	 * This is more efficient than retrieving the whole content, or retrieving parts in separate
84  	 * calls, because the file is only opened once and random access is used to read only the
85  	 * required parts.
86  	 *
87  	 * NOTE: if offset and length are both -1, retrieves the whole content. This is used by the
88  	 * retrieve(id) method.
89  	 *
90  	 * NOTE2: Mapped file IO on Windows has some issues that sometimes cause an OutOfMemoryError on
91  	 * the FileChannel.map() call (which makes no sense, because memory mapping only uses address
92  	 * space, it doesn't try to read the whole file). Possibly this could be solved by using 64-bit
93  	 * Java, but we haven't tried. For now we just disable memory mapping on Windows.
94  	 *
95  	 * @param fiid
96  	 *            forward index document id
97  	 * @param start
98  	 *            the starting points of the substrings (in words)
99  	 * @param end
100 	 *            the end points (i.e. first token beyond) of the substrings (in words)
101 	 * @return the parts
102 	 * @deprecated use retrievePartsInt and getTerms().get(id)
103 	 */
104 	@Deprecated
105 	public synchronized List<String[]> retrieveParts(int fiid, int[] start, int[] end) {
106 
107 		// First, retrieve the token ids
108 		List<int[]> resultInt = retrievePartsInt(fiid, start, end);
109 
110 		// Translate them to strings using the terms index
111 		List<String[]> result = new ArrayList<>(resultInt.size());
112 		for (int[] snippetInt: resultInt) {
113 			String[] snippet = new String[snippetInt.length];
114 			for (int j = 0; j < snippetInt.length; j++) {
115 				snippet[j] = getTerms().get(snippetInt[j]);
116 			}
117 			result.add(snippet);
118 		}
119 		return result;
120 	}

217 	/**
218 	 * Open a forward index.
219 	 *
220 	 * Automatically figures out the forward index version and
221 	 * instantiates the right class.
222 	 *
223 	 * @param dir forward index directory
224 	 * @param indexMode true iff we're in index mode (writing to
225 	 *   the forward index); otherwise it will be read-only.
226 	 * @param collator collator to use for sorting
227 	 * @param create if true, create a new forward index
228 	 * @return the forward index object
229 	 */
230 	public static ForwardIndex open(File dir, boolean indexMode, Collator collator, boolean create) {
231 
232 		if (!dir.exists()) {
233 			if (!create)
234 				throw new IllegalArgumentException("ForwardIndex doesn't exist: " + dir);
235 			dir.mkdir();
236 		}
237 
238 		// Version check
239 		String version = CURRENT_VERSION;
240 		if (!indexMode || !create) {
241 			// We're opening an existing forward index. Check version.
242 			if (!VersionFile.isTypeVersion(dir, "fi", CURRENT_VERSION)) {
243 				if (VersionFile.isTypeVersion(dir, "fi", "3")) {
244 					version = "3";
245 				} else if (VersionFile.isTypeVersion(dir, "fi", "2")) {
246 					version = "2";
247 				} else {
248 					throw new IllegalArgumentException("Not a forward index or wrong version: "
249 							+ VersionFile.report(dir) + " (fi " + CURRENT_VERSION + " expected)");
250 				}
251 			}
252 		} else {
253 			// We're creating a forward index. Write version.
254 			VersionFile.write(dir, "fi", CURRENT_VERSION);
255 		}
256 
257 		ForwardIndex fi;
258 		boolean largeTermsFileSupport = true;
259 		switch(version) {
260 		case "2":
261 			throw new UnsupportedOperationException("Forward index version (2) too old for this BlackLab version. Please re-index.");
262 		case "3":
263 			largeTermsFileSupport = false;
264 			break;
265 		case "4":
266 			break;
267 		}
268 		fi = new ForwardIndexImplV3(dir, indexMode, collator, create, largeTermsFileSupport);
269 		return fi;
270 	}

286 	/** Perform a task on each document in the forward index.
287 	 * @param task the task to perform
288 	 */
289 	public void forEachDocument(ForwardIndexDocTask task) {
290 		for (Integer fiid: idSet()) {
291 			int[] tokenIds = retrievePartsInt(fiid, new int[] {-1}, new int[] {-1}).get(0);
292 			task.perform(fiid, tokenIds);
293 		}
