157 	@Override
158 	public void setIdTranslateInfo(IndexReader reader, String lucenePropFieldName) {
159 		this.reader = reader;
160 		this.fiidFieldName = ComplexFieldUtil.forwardIndexIdField(lucenePropFieldName);
161 		try {
162 			LeafReader srw = SlowCompositeReaderWrapper.wrap(reader);
163 			Map<String, UninvertingReader.Type> fields = new HashMap<>();
164 			fields.put(fiidFieldName, UninvertingReader.Type.INTEGER);
165 			uninv = new UninvertingReader(srw, fields);
166 			cachedFiids = uninv.getNumericDocValues(fiidFieldName);
167 
168 			// Check if the cache was retrieved OK
169 			boolean allZeroes = true;
170 			int numToCheck = Math.min(NUMBER_OF_CACHE_ENTRIES_TO_CHECK, srw.maxDoc());
171 			for (int i = 0; i < numToCheck; i++) {
172 				// (NOTE: we don't check if document wasn't deleted, but that shouldn't matter here)
173 				if (cachedFiids.get(i) != 0) {
174 					allZeroes = false;
175 					break;
176 				}
177 			}
178 			if (allZeroes) {
179 				// Tokens lengths weren't saved in the index, skip cache
180 				cachedFiids = null;
181 			}
182 
183 		} catch (IOException e) {
184 			throw new RuntimeException(e);
185 		}
186 	}

188 	@Override
189 	public int luceneDocIdToFiid(int docId) {
190 		if (cachedFiids != null)
191 			return (int)cachedFiids.get(docId);
192 
193 		// Not cached; find fiid by reading stored value from Document now
194 		try {
195 			return Integer.parseInt(reader.document(docId).get(fiidFieldName));
196 		} catch (Exception e) {
197 			throw new RuntimeException(e);
198 		}
199 	}

201 	ForwardIndexImplV3(File dir, boolean indexMode, Collator collator, boolean create, boolean largeTermsFileSupport) {
202 		if (!dir.exists()) {
203 			if (!create)
204 				throw new IllegalArgumentException("ForwardIndex doesn't exist: " + dir);
205 			dir.mkdir();
206 		}
207 
208 		this.indexMode = indexMode;
209 
210 		termsFile = new File(dir, "terms.dat");
211 		tocFile = new File(dir, "docs.dat");
212 		tokensFile = new File(dir, "tokens.dat");
213 		if (create) {
214 			if (tokensFile.exists())
215 				tokensFile.delete();
216 			if (tocFile.exists())
217 				tocFile.delete();
218 			if (termsFile.exists())
219 				termsFile.delete();
220 		}
221 		toc = new ArrayList<>();
222 		deletedTocEntries = new ArrayList<>();
223 		try {
224 			setLargeTermsFileSupport(largeTermsFileSupport);
225 			boolean existing = false;
226 			if (tocFile.exists()) {
227 				readToc();
228 				terms = new TermsImplV3(indexMode, collator, termsFile, useBlockBasedTermsFile);
229 				existing = true;
230 				tocModified = false;
231 			} else {
232 				terms = new TermsImplV3(indexMode, collator, null, true);
233 				tokensFile.createNewFile();
234 				tokensFileChunks = null;
235 				tocModified = true;
236 				terms.setBlockBasedFile(useBlockBasedTermsFile);
237 			}
238 			openTokensFile();
239 
240 			// Tricks to speed up reading
241 			if (existing && !create) {
242 				if (!indexMode && useMemoryMapping) {
243 
244 					// Memory-map the file
245 					// NOTE: We only use this in search mode right now.
246 					// @@@ add support for mapped write? (need to re-map as the file grows in size)
247 					memoryMapTokensFile(false);
248 
249 				} else {
250 					// Don't use memory mapping. Just read from file channel.
251 				}
252 			}
253 
254 		} catch (IOException e) {
255 			throw new RuntimeException(e);
256 		}
257 
258 		if (create) {
259 			clear();
260 		}
261 	}

265 		tokensFileChannel = tokensFp.getChannel();

268 	private void memoryMapTokensFile(boolean keepInMemory) throws IOException {
269 
270 		// Map the tokens file in chunks of 2GB each. When retrieving documents, we always
271 		// read it from just one chunk, not multiple, but because each chunk begins at a
272 		// document start, documents of up to 2G tokens can be processed. We could get around
273 		// this limitation by reading from multiple chunks, but this would make the code
274 		// more complex.
275 		tokensFileChunks = new ArrayList<>();
276 		tokensFileChunkOffsetBytes = new ArrayList<>();
277 		long mappedBytes = 0;
278 		long tokenFileEndBytes = tokenFileEndPosition * SIZEOF_INT;
279 		while (mappedBytes < tokenFileEndBytes) {
280 			// Find the last TOC entry start point that's also in the previous mapping
281 			// (or right the first byte after the previous mapping).
282 
283 			// Look for the largest entryOffset that's no larger than mappedBytes.
284 			TocEntry mapNextChunkFrom = null;
285 			for (TocEntry e: toc) {
286 				if (e.offset <= mappedBytes && (mapNextChunkFrom == null || e.offset > mapNextChunkFrom.offset))
287 					mapNextChunkFrom = e;
288 			}
289 
290 			// Uses binary search.
291 			int min = 0, max = toc.size();
292 			while (max - min > 1) {
293 				int middle = (min + max) / 2;
294 				long middleVal = toc.get(middle).offset * SIZEOF_INT;
295 				if (middleVal <= mappedBytes) {
296 					min = middle;
297 				} else {
298 					max = middle;
299 				}
300 			}
301 			long startOfNextMappingBytes = toc.get(min).offset * SIZEOF_INT;
302 
303 			// Map this chunk
304 			long sizeBytes = tokenFileEndBytes - startOfNextMappingBytes;
305 			if (sizeBytes > preferredChunkSizeBytes)
306 				sizeBytes = preferredChunkSizeBytes;
307 
308 			ByteBuffer mapping;
309 			if (keepInMemory) {
310 				mapping = ByteBuffer.allocate((int) sizeBytes);
311 				tokensFileChannel.position(startOfNextMappingBytes);
312 				int bytesRead = tokensFileChannel.read(mapping);
313 				if (bytesRead != mapping.capacity()) {
314 					throw new RuntimeException("Could not read tokens file chunk into memory!");
315 				}
316 			} else {
317 				mapping = tokensFileChannel.map(FileChannel.MapMode.READ_ONLY,
318 						startOfNextMappingBytes, sizeBytes);
319 			}
320 			tokensFileChunks.add(mapping);
321 			tokensFileChunkOffsetBytes.add(startOfNextMappingBytes);
322 			mappedBytes = startOfNextMappingBytes + sizeBytes;
323 		}
324 	}

327 	 * Delete all content in the forward index
328 	 */
329 	private void clear() {
330 		if (!indexMode)
331 			throw new RuntimeException("Cannot clear, not in index mode");
332 
333 		// delete data files and empty TOC
334 		try {
335 			if (tokensFp == null)
336 				openTokensFile();
337 			tokensFp.setLength(0);
338 		} catch (IOException e) {
339 			throw new RuntimeException(e);
340 		}
341 		termsFile.delete();
342 		tocFile.delete();
343 		toc.clear();
344 		deletedTocEntries.clear();
345 		tokenFileEndPosition = 0;
346 		tocModified = true;
347 	}

349 	/**
350 	 * Read the table of contents from the file
351 	 */
352 	private void readToc() {
353 		toc.clear();
354 		deletedTocEntries.clear();
355 		try (RandomAccessFile raf = new RandomAccessFile(tocFile, "r");
356 			FileChannel fc = raf.getChannel()) {
357 			long fileSize = tocFile.length();
358 			MappedByteBuffer buf = fc.map(MapMode.READ_ONLY, 0, fileSize);
359 			int n = buf.getInt();
360 			long[] offset = new long[n];
361 			int[] length = new int[n];
362 			byte[] deleted = new byte[n];
363 			LongBuffer lb = buf.asLongBuffer();
364 			lb.get(offset);
365 			buf.position(buf.position() + SIZEOF_LONG * n);
366 			IntBuffer ib = buf.asIntBuffer();
367 			ib.get(length);
368 			buf.position(buf.position() + SIZEOF_INT * n);
369 			buf.get(deleted);
370 			toc.ensureCapacity(n);
371 			for (int i = 0; i < n; i++) {
372 				TocEntry e = new TocEntry(offset[i], length[i], deleted[i] != 0);
373 				toc.add(e);
374 				if (e.deleted) {
375 					deletedTocEntries.add(e);
376 				}
377 				long end = e.offset + e.length;
378 				if (end > tokenFileEndPosition)
379 					tokenFileEndPosition = end;
380 			}
381 			sortDeletedTocEntries();
382 		} catch (Exception e) {
383 			throw ExUtil.wrapRuntimeException(e);
384 		}
385 		toc.trimToSize();
386 		deletedTocEntries.trimToSize();
387 	}
388 
389 	private void sortDeletedTocEntries() {
390 		Collections.sort(deletedTocEntries, new Comparator<TocEntry>() {
391 			@Override
392 			public int compare(TocEntry o1, TocEntry o2) {
393 				return o1.length - o2.length;
394 			}
395 		});
396 	}

398 	/**
399 	 * Write the table of contents to the file
400 	 */
401 	private void writeToc() {
402 
403 		if (!indexMode)
404 			throw new RuntimeException("Cannot write ToC, not in index mode");
405 
406 		try {
407 			int n = toc.size();
408 			long[] offset = new long[n];
409 			int[] length = new int[n];
410 			byte[] deleted = new byte[n];
411 			int i = 0;
412 			for (TocEntry e: toc) {
413 				offset[i] = e.offset;
414 				length[i] = e.length;
415 				deleted[i] = (byte) (e.deleted ? 1 : 0);
416 				i++;
417 			}
418 			try (RandomAccessFile raf = new RandomAccessFile(tocFile, "rw")) {
419 				try (FileChannel fc = raf.getChannel()) {
420 					long fileSize = SIZEOF_INT + (SIZEOF_LONG + SIZEOF_INT + 1) * n;
421 					fc.truncate(fileSize);
422 					MappedByteBuffer buf = fc.map(MapMode.READ_WRITE, 0, fileSize);
423 					buf.putInt(n);
424 					LongBuffer lb = buf.asLongBuffer();
425 					lb.put(offset);
426 					buf.position(buf.position() + SIZEOF_LONG * n);
427 					IntBuffer ib = buf.asIntBuffer();
428 					ib.put(length);
429 					buf.position(buf.position() + SIZEOF_INT * n);
430 					buf.put(deleted);
431 				}
432 			}
433 		} catch (Exception e) {
434 			throw ExUtil.wrapRuntimeException(e);
435 		}
436 		tocModified = false;
437 	}

440 	public void close() {
441 		try {
442 			if (tocModified) {
443 				writeToc();
444 				terms.write(termsFile);
445 			}
446 
447 			if (tokensFileChannel != null) {
448 				// Cannot truncate if still mapped; cannot force demapping.
449 				//tokensFileChannel.truncate(tokenFileEndPosition * SIZEOF_INT);
450 				tokensFileChannel.close();
451 			}
452 
453 			if (tokensFp != null)
454 				tokensFp.close();
455 
456 		} catch (Exception e) {
457 			throw ExUtil.wrapRuntimeException(e);
458 		}
459 	}

461 	/**
462 	 * Find the best-fitting deleted entry for the specified length
463 	 * @param length length the entry should at least be
464 	 * @return the best-fitting entry
465 	 */
466 	TocEntry findBestFittingGap(int length) {
467 		int n = deletedTocEntries.size();
468 
469 		// Are there any fitting gaps?
470 		if (n == 0 || deletedTocEntries.get(n - 1).length < length)
471 			return null;
472 
473 		// Does the smallest gap fit?
474 		if (deletedTocEntries.get(0).length >= length)
475 			return deletedTocEntries.get(0);
476 
477 		// Do a binary search to find the best fit
478 		int doesntFit = 0, bestFitSoFar = n - 1;
479 		while (bestFitSoFar - doesntFit > 1) {
480 			int newTry = doesntFit + bestFitSoFar / 2;
481 			if (deletedTocEntries.get(newTry).length < length)
482 				doesntFit = newTry;
483 			else
484 				bestFitSoFar = newTry;
485 		}
486 		return deletedTocEntries.get(bestFitSoFar);
487 	}

489 	@Override
490 	public synchronized int addDocument(List<String> content, List<Integer> posIncr) {
491 		if (!indexMode)
492 			throw new RuntimeException("Cannot add document, not in index mode");
493 
494 		// Calculate the total number of tokens we need to store, based on the number
495 		// of positions (we store 1 token per position, regardless of whether we have
496 		// none, one or multiple values for that position)
497 		int numberOfTokens;
498 		if (posIncr == null) {
499 			// No position increments given; assume always 1
500 			numberOfTokens = content.size();
501 		} else {
502 			// Calculate using position increments
503 			numberOfTokens = 0;
504 			for (int inc: posIncr) {
505 				numberOfTokens += inc;
506 			}
507 		}
508 
509 		// Decide where we're going to store this document,
510 		// and update ToC
511 		TocEntry gap = findBestFittingGap(numberOfTokens);
512 		long newDocumentOffset;
513 		int mapReserve;
514 		tocModified = true;
515 		boolean addNewEntry = true;
516 		int newDocumentFiid = -1;
517 		if (gap == null) {
518 			// No fitting gap; just write it at the end
519 			newDocumentOffset = tokenFileEndPosition;
520 			mapReserve = WRITE_MAP_RESERVE; // if writing at end, reserve more space
521 		}
522 		else {
523 			// Found a fitting gap; write it there
524 			newDocumentOffset = gap.offset;
525 			mapReserve = 0; // don't reserve extra write space, not needed
526 			if (gap.length == numberOfTokens) {
527 				// Exact fit; delete from free list and re-use entry
528 				deletedTocEntries.remove(gap);
529 				gap.deleted = false;
530 				addNewEntry = false;
531 				newDocumentFiid = toc.indexOf(gap);
532 			} else {
533 				// Not an exact fit; calculate remaining gap and re-sort free list
534 				gap.offset += numberOfTokens;
535 				gap.length -= numberOfTokens;
536 				sortDeletedTocEntries();
537 			}
538 		}
539 		// Do we need to create a new entry for this document in the ToC?
540 		// (always, unless we found an exact-fitting gap)
541 		if (addNewEntry) {
542 			// See if there's an unused entry
543 			TocEntry smallestFreeEntry = deletedTocEntries.isEmpty() ? null : deletedTocEntries.get(0);
544 			if (smallestFreeEntry != null && smallestFreeEntry.length == 0) {
545 				// Yes; re-use
546 				deletedTocEntries.remove(0);
547 				smallestFreeEntry.offset = newDocumentOffset;
548 				smallestFreeEntry.length = numberOfTokens;
549 				smallestFreeEntry.deleted = false;
550 				newDocumentFiid = toc.indexOf(smallestFreeEntry);
551 			} else {
552 				// No; make new entry
553 				toc.add(new TocEntry(newDocumentOffset, numberOfTokens, false));
554 				newDocumentFiid = toc.size() - 1;
555 			}
556 		}
557 
558 		try {
559 			// Can we use the current write buffer for this write?
560 			long writeBufEnd = writeBuffer == null ? 0 : writeBufOffset + writeBuffer.limit();
561 			if (writeBuffer == null || writeBufOffset > newDocumentOffset || writeBufEnd < newDocumentOffset + numberOfTokens) {
562 				// No, remap it
563 				writeBufOffset = newDocumentOffset;
564 				MappedByteBuffer byteBuffer = tokensFileChannel.map(FileChannel.MapMode.READ_WRITE,
565 						writeBufOffset * SIZEOF_INT, (numberOfTokens + mapReserve)
566 								* SIZEOF_INT);
567 				writeBuffer = byteBuffer.asIntBuffer();
568 			}
569 
570 			// Set the correct start position
571 			writeBuffer.position((int)(newDocumentOffset - writeBufOffset));
572 
573 			// Did we increase the length of the tokens file?
574 			long end = newDocumentOffset + numberOfTokens;
575 			if (end > tokenFileEndPosition)
576 				tokenFileEndPosition = end;
577 
578 			// Write the token ids
579 			// (first fill the buffer, then write the buffer in 1 call)
580 			int [] tokenIds = new int[numberOfTokens];
581 			int tokenIdsIndex = 0;
582 			Iterator<String> contentIt = content.iterator();
583 			Iterator<Integer> posIncrIt = posIncr == null ? null : posIncr.iterator();
584 			int emptyStringTokenId = posIncrIt != null ? terms.indexOf("") : -1;
585 			while (contentIt.hasNext()) {
586 				String token = contentIt.next();
587 				int pi = posIncrIt == null ? 1 : posIncrIt.next();
588 				if (pi == 0)
589 					continue; // we only store the first token at any position
590 				if (pi > 1) {
591 					// Skipped a few tokens; add empty tokens for these positions
592 					for (int i = 0; i < pi - 1; i++) {
593 						tokenIds[tokenIdsIndex] = emptyStringTokenId;
594 						tokenIdsIndex++;
595 					}
596 				}
597 
598 				tokenIds[tokenIdsIndex] = terms.indexOf(token);
599 				tokenIdsIndex++;
600 			}
601 			if (tokenIdsIndex != numberOfTokens)
602 				throw new RuntimeException("tokenIdsIndex != numberOfTokens (" + tokenIdsIndex + " != " + numberOfTokens + ")");
603 			writeBuffer.put(tokenIds);
604 
605 			return newDocumentFiid;
606 		} catch (IOException e1) {
607 			throw new RuntimeException(e1);
608 		}
609 	}

611 	@Override
612 	public synchronized List<int[]> retrievePartsInt(int fiid, int[] start, int[] end) {
613 		try {
614 			TocEntry e = toc.get(fiid);
615 			if (e == null || e.deleted)
616 				return null;
617 
618 			int n = start.length;
619 			if (n != end.length)
620 				throw new IllegalArgumentException("start and end must be of equal length");
621 			List<int[]> result = new ArrayList<>(n);
622 
623 			for (int i = 0; i < n; i++) {
624 				if (start[i] == -1)
625 					start[i] = 0;
626 				if (end[i] == -1)
627 					end[i] = e.length;
628 				if (start[i] < 0 || end[i] < 0) {
629 					throw new IllegalArgumentException("Illegal values, start = " + start[i] + ", end = "
630 							+ end[i]);
631 				}
632 				if (end[i] > e.length) // Can happen while making KWICs because we don't know the
633 										// doc length until here
634 					end[i] = e.length;
635 				if (start[i] > e.length || end[i] > e.length) {
636 					throw new IllegalArgumentException("Value(s) out of range, start = " + start[i]
637 							+ ", end = " + end[i] + ", content length = " + e.length);
638 				}
639 				if (end[i] <= start[i]) {
640 					throw new IllegalArgumentException(
641 							"Tried to read empty or negative length snippet (from " + start[i]
642 									+ " to " + end[i] + ")");
643 				}
644 
645 				// Get an IntBuffer to read the desired content
646 				IntBuffer ib = null;
647 				boolean inMem = false;
648 				if (tokensFileChunks != null) {
649 					// Yes, the tokens file has either been fully loaded into memory or
650 					// is mapped into memory. Get an int buffer into the file.
651 					inMem = true;
652 
653 					// Figure out which chunk to access.
654 					ByteBuffer whichChunk = null;
655 					long chunkOffsetBytes = -1;
656 					long entryOffsetBytes = e.offset * SIZEOF_INT;
657 					for (int j = 0; j < tokensFileChunkOffsetBytes.size(); j++) {
658 						long offsetBytes = tokensFileChunkOffsetBytes.get(j);
659 						ByteBuffer buffer = tokensFileChunks.get(j);
660 						if (offsetBytes <= entryOffsetBytes + start[i] * SIZEOF_INT
661 								&& offsetBytes + buffer.capacity() >= entryOffsetBytes + end[i]
662 										* SIZEOF_INT) {
663 							// This one!
664 							whichChunk = buffer;
665 							chunkOffsetBytes = offsetBytes;
666 							break;
667 						}
668 					}
669 
670 					whichChunk.position((int) (e.offset * SIZEOF_INT - chunkOffsetBytes));
671 					ib = whichChunk.asIntBuffer();
672 				}
673 
674 				int snippetLength = end[i] - start[i];
675 				int[] snippet = new int[snippetLength];
676 				if (TESTING_IO_IMPACT) {
677 					// We're testing how much impact forward index I/O has on sorting/grouping.
678 					// Fill the array with random token ids instead of reading them from the
679 					// file.
680 					int numberOfTerms = terms.numberOfTerms();
681 					for (int j = 0; j < snippetLength; j++) {
682 						int randomTermId = (int) Math.random() * numberOfTerms;
683 						snippet[j] = randomTermId;
684 					}
685 				} else {
686 					if (inMem) {
687 						// The file is mem-mapped.
688 						// Position us at the correct place in the file.
689 						ib.position(start[i]);
690 					} else {
691 						// Not mapped. Explicitly read the part we require from disk into an int
692 						// buffer.
693 						long offset = e.offset + start[i];
694 
695 						int bytesToRead = snippetLength * SIZEOF_INT;
696 						ByteBuffer buffer = ByteBuffer.allocate(bytesToRead);
697 						int bytesRead = tokensFileChannel.read(buffer, offset * SIZEOF_INT);
698 						if (bytesRead < bytesToRead) {
699 							throw new RuntimeException("Not enough bytes read: " + bytesRead
700 									+ " < " + bytesToRead);
701 						}
702 						buffer.position(0);
703 						ib = buffer.asIntBuffer();
704 					}
705 					ib.get(snippet);
706 				}
707 				result.add(snippet);
708 			}
709 
710 			return result;
711 		} catch (Exception e) {
712 			throw ExUtil.wrapRuntimeException(e);
713 		}
714 	}

731 	@Override
732 	public void deleteDocument(int fiid) {
733 		if (!indexMode)
734 			throw new RuntimeException("Cannot delete document, not in index mode");
735 		TocEntry tocEntry = toc.get(fiid);
736 		tocEntry.deleted = true;
737 		deletedTocEntries.add(tocEntry); // NOTE: mergeAdjacentDeletedEntries takes care of re-sorting
738 		mergeAdjacentDeletedEntries();
739 		tocModified = true;
740 	}
741 
742 	/**
743 	 * Check if we can merge two (or more) deleted entries to create a
744 	 * larger gap, and do so.
745 	 *
746 	 * Also takes care of truncating the file if there are deleted entries
747 	 * at the end.
748 	 */
749 	private void mergeAdjacentDeletedEntries() {
750 		// Sort by offset, so we can find adjacent entries
751 		Collections.sort(deletedTocEntries);
752 
753 		// Find and merge adjacent entries
754 		TocEntry prev = deletedTocEntries.get(0);
755 		for (int i = 1; i < deletedTocEntries.size(); i++) {
756 			TocEntry current = deletedTocEntries.get(i);
757 			if (current.offset == prev.offset + prev.length) {
758 				// Found two adjacent deleted entries. Merge them.
759 				current.offset = prev.offset;
760 				current.length += prev.length;
761 
762 				// length == 0 means a toc entry is unused
763 				// we can't delete toc entries because it messes up
764 				// the fiids. We will reuse them in addDocument().
765 				prev.length = 0;
766 			}
767 			prev = current;
768 		}
769 
770 		TocEntry lastEntry = deletedTocEntries.get(deletedTocEntries.size() - 1);
771 		if (lastEntry.offset + lastEntry.length >= tokenFileEndPosition) {
772 			// Free entry at the end of the token file. Remove the entry and
773 			// make the tokens file shorter.
774 			tokenFileEndPosition -= lastEntry.length;
775 			lastEntry.length = 0;
776 		}
777 
778 		// Re-sort on gap length
779 		sortDeletedTocEntries();
780 	}

806 	@Override

808 		return new AbstractSet<Integer>() {
809 			@Override
810 			public boolean contains(Object o) {
811 				return !toc.get((Integer)o).deleted;
812 			}
813 
814 			@Override
815 			public boolean isEmpty() {
816 				return toc.size() == deletedTocEntries.size();
817 			}
818 
819 			@Override
820 			public Iterator<Integer> iterator() {
821 				return new Iterator<Integer>() {
822 					int current = -1;
823 					int next = -1;
824 
825 					@Override
826 					public boolean hasNext() {
827 						if (next < 0)
828 							findNext();
829 						return next < toc.size();
830 					}
831 
832 					private void findNext() {
833 						next = current + 1;
834 						while (next < toc.size() && toc.get(next).deleted) {
835 							next++;
836 						}
837 					}
838 
839 					@Override
840 					public Integer next() {
841 						if (next < 0)
842 							findNext();
843 						if (next >= toc.size())
844 							throw new NoSuchElementException();
845 						current = next;
846 						next = -1;
847 						return current;
848 					}
849 
850 					@Override
851 					public void remove() {
852 						throw new UnsupportedOperationException();
853 					}
854 				};
855 			}
856 
857 			@Override
858 			public int size() {
859 				return toc.size() - deletedTocEntries.size();
860 			}
861 		};
