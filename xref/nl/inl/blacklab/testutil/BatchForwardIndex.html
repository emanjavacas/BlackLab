9   import org.apache.log4j.BasicConfigurator;

11  /**
12   * Executes a batch of fetch operations on a forward index.
13   */
14  public class BatchForwardIndex {
15  
16  	private static final int SNIPPET_LENGTH_WORDS = 11;
17  
18  	public static void main(String[] args) throws Exception {
19  
20  		BasicConfigurator.configure(); // suppress log4j warning
21  
22  		int fileArgNumber = 0;
23  		File indexDir = null, inputFile = null;
24  		for (int i = 0; i < args.length; i++) {
25  			String arg = args[i].trim();
26  			if (arg.charAt(0) == '-') {
27  				System.err.println("Illegal option: " + arg);
28  				usage();
29  				return;
30  			}
31  			switch (fileArgNumber) {
32  			case 0:
33  				indexDir = new File(arg);
34  				if (!indexDir.exists() || !indexDir.isDirectory()) {
35  					System.err.println("Index directory not found: " + arg);
36  					usage();
37  					return;
38  				}
39  				break;
40  			case 1:
41  				inputFile = new File(arg);
42  				if (!inputFile.exists()) {
43  					System.err.println("Input file not found: " + arg);
44  					usage();
45  					return;
46  				}
47  				break;
48  			default:
49  				System.err.println("Too many file arguments (supply index dir and input file)");
50  				usage();
51  				return;
52  			}
53  			fileArgNumber++;
54  		}
55  		if (fileArgNumber < 2) {
56  			System.err.println("Too few file arguments (supply index dir and input file)");
57  			usage();
58  			return;
59  		}
60  
61  		System.err.print("Opening forward index... ");
62  		ForwardIndex fi = ForwardIndex.open(indexDir, false, null, false);
63  		System.err.println("done. [#docs: " + fi.getNumDocs() + "]");
64  
65  		System.out.println("First\tNumber\tSkip\tSnippets\tTime");
66  
67  		for (String query : FileUtil.readLines(inputFile)) {
68  			query = query.trim();
69  			if (query.length() == 0 || query.charAt(0) == '#')
70  				continue; // skip empty lines and #-comments
71  			String[] numberStr = query.split("\\s+");
72  			int[] numbers = new int[numberStr.length];
73  			try {
74  				for (int i = 0; i < numberStr.length; i++) {
75  					numbers[i] = Integer.parseInt(numberStr[i]);
76  				}
77  
78  				int first = numbers[0];
79  				int number = numbers.length > 1 ? numbers[1] : 100;
80  				int skip = numbers.length > 2 ? numbers[2] : 0;
81  				int snippets = numbers.length > 3 ? numbers[3] : 5;
82  				long time = doPerformanceTest(fi, first, number, skip, snippets);
83  				System.out.println(String.format("%d\t%d\t%d\t%d\t%d", first, number, skip, snippets,
84  						time));
85  
86  			} catch (Exception e) {
87  				e.printStackTrace();
88  				System.err.println("Error with line '" + query + "'; skipping...");
89  			}
90  
91  		}
92  	}
93  
94  	/**
95  	 * Do a performance test, retrieving snippets from a number of documents.
96  	 * @param fi the forward index
97  	 *
98  	 * @param first
99  	 *            fiid (position in toc) of first document to access
100 	 * @param number
101 	 *            number of documents to access
102 	 * @param skip
103 	 *            number of documents to skip between accesses
104 	 * @param snippets
105 	 *            number of random snippets to retrieve from each document
106 	 * @return elapsed time in ms
107 	 */
108 	public static long doPerformanceTest(ForwardIndex fi, int first, int number, int skip,
109 			int snippets) {
110 		Timer t = new Timer();
111 		int[] start = new int[snippets];
112 		int[] end = new int[snippets];
113 		int docPos = first;
114 		for (int i = 0; i < number; i++) {
115 			int length;
116 			do {
117 				if (docPos >= fi.getNumDocs())
118 					throw new RuntimeException("Performance test went beyond end of forward index ("
119 							+ fi.getNumDocs() + " docs)");
120 
121 				// Choose random snippets
122 				length = fi.getDocLength(docPos);
123 				if (length == 0) // can't get snippet from empty doc
124 					docPos++;
125 			} while (length == 0);
126 			int snippetLength = Math.min(SNIPPET_LENGTH_WORDS, length);
127 			for (int j = 0; j < snippets; j++) {
128 				start[j] = (int) (Math.random() * (length - snippetLength));
129 				end[j] = start[j] + snippetLength;
130 			}
131 
132 			// Retrieve snippets
133 			fi.retrievePartsInt(docPos, start, end);
134 
135 			// Go to next doc
136 			docPos += skip + 1;
137 		}
138 		return t.elapsed();
139 	}
140 
141 	private static void usage() {
142 		System.err.println("\nUsage: " + BatchForwardIndex.class.getSimpleName()
143 				+ " [options] <forwardIndexDir> <inputfile>\n\n"
144 				+ "<inputfile> should contain lines of whitespace-separated integers:\n"
145 				+ "   <first> <number> <skip> <snippets>\n" + "\n"
146 				+ "   first: position of the first document to access\n"
147 				+ "   number: number of documents to access [100]\n"
148 				+ "   skip: how many documents to skip between accesses [0]\n"
149 				+ "   snippets: how many random snippets to retrieve per document [5]\n" + "\n"
150 				+ "Options:\n"
151 				+ "-m don't try to read ForwardIndex into memory\n"
152 				+ "\n"
153 				+ "Output:\n" + "<first> <number> <skip> <snippets>\t<searchTimeMs>\n");
154 	}
155 }
